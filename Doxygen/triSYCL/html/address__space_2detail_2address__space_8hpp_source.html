114/** Dispatch the address space implementation according to the requested type
115
116 \param T is the type of the object to be created
117
118 \param AS is the address space to place the object into or to point to
119 in the case of a pointer type
120*/
121template <typename T, access::address_space AS>
122using addr_space =
123 typename std::conditional<std::is_pointer<T>::value,
124 address_space_ptr<T, AS>,
125 typename std::conditional<std::is_class<T>::value,
126 address_space_object<T, AS>,
127 typename std::conditional<std::is_array<T>::value,
128 address_space_array<T, AS>,
129 address_space_fundamental<T, AS>
130 >::type>::type>::type;

133/** Implementation of the base infrastructure to wrap something in an
134 OpenCL address space
135
136 \param T is the type of the basic stuff to be created
137
138 \param AS is the address space to place the object into
139
140 \todo Verify/improve to deal with const/volatile?
141*/
142template <typename T, access::address_space AS>
143struct address_space_base {
144 /** Store the base type of the object
145
146 \todo Add to the specification
147 */
148 using type = T;
149
150 /** Store the base type of the object with OpenCL address space modifier
151
152 \todo Add to the specification
153 */
154 using opencl_type = typename ocl_type<T, AS>::type;
155
156 /** Set the address_space identifier that can be queried to know the
157 pointer type */
158 static auto constexpr address_space = AS;

163/** Implementation of a variable with an OpenCL address space
164
165 \param T is the type of the basic object to be created
166
167 \param AS is the address space to place the object into
168*/
169template <typename T, access::address_space AS>
170struct address_space_variable : public address_space_base<T, AS> {
171 /** Store the base type of the object with OpenCL address space modifier
172
173 \todo Add to the specification
174 */
175 using opencl_type = typename ocl_type<T, AS>::type;
176
177 /// Keep track of the base class as a short-cut
178 using super = address_space_base<T, AS>;
179
180protected:
181
182 /* C++11 helps a lot to be able to have the same constructors as the
183 parent class here
184
185 \todo Add this to the list of required C++11 features needed for SYCL
186 */
187 opencl_type variable;
188
189public:
190
191 /** Allow to create an address space version of an object or to convert
192 one to be used by the classes inheriting by this one because it is
193 not possible to directly initialize a base class member in C++ */
194 address_space_variable(const T & v) : variable(v) { }
195
196
197 /// Put back the default constructors canceled by the previous definition
198 address_space_variable() = default;
199
200
201 /** Conversion operator to allow a address_space_object<T> to be used
202 as a T so that all the methods of a T and the built-in operators for
203 T can be used on a address_space_object<T> too.
204
205 Use opencl_type so that if we take the address of it, the address
206 space is kept.
207 */
208 operator opencl_type & () { return variable; }
209
210 /// Return the address of the value to implement pointers
211 opencl_type * get_address() { return &variable; }
212
213};

216/** Implementation of a fundamental type with an OpenCL address space
217
218 \param T is the type of the basic object to be created
219
220 \param AS is the address space to place the object into
221
222 \todo Verify/improve to deal with const/volatile?
223*/
224template <typename T, access::address_space AS>
225struct address_space_fundamental : public address_space_variable<T, AS> {
226 /// Keep track of the base class as a short-cut
227 using super = address_space_variable<T, AS>;
228
229 /// Inherit from base class constructors
230 using super::address_space_variable;
231
232
233 /** Also request for the default constructors that have been disabled by
234 the declaration of another constructor
235
236 This ensures for example that we can write
237 \code
238 generic<float *> q;
239 \endcode
240 without initialization.
241 */
242 address_space_fundamental() = default;
243
244
245 /** Allow for example assignment of a global<float> to a priv<double>
246 for example
247
248 Since it needs 2 implicit conversions, it does not work with the
249 conversion operators already define, so add 1 more explicit
250 conversion here so that the remaining implicit conversion can be
251 found by the compiler.
252
253 Strangely
254 \code
255 template <typename SomeType, address_space SomeAS>
256 address_space_base(addr_space<SomeType, SomeAS>& v)
257 : variable(SomeType(v)) { }
258 \endcode
259 cannot be used here because SomeType cannot be inferred. So use
260 address_space_base<> instead
261
262 Need to think further about it...
263 */
264 template <typename SomeType, trisycl::access::address_space SomeAS>
265 address_space_fundamental(address_space_fundamental<SomeType, SomeAS>& v)
266 {
267 /* Strangely I cannot have it working in the initializer instead, for
268 some cases */
269 super::variable = SomeType(v);
270 }
271
272};

275/** Implementation for an OpenCL address space pointer
276
277 \param T is the pointer type
278
279 Note that if \a T is not a pointer type, it is an error.
280
281 All the address space pointers inherit from it, which makes trivial
282 the implementation of \c trisycl::multi_ptr<T, AS>
283*/
284template <typename T, access::address_space AS>
285struct address_space_ptr : public address_space_fundamental<T, AS> {
286 // Verify that \a T is really a pointer
287 static_assert(std::is_pointer<T>::value,
288 "T must be a pointer type");
289
290 /// Keep track of the base class as a short-cut
291 using super = address_space_fundamental<T, AS>;
292
293 /// Inherit from base class constructors
294 using super::address_space_fundamental;
295
296 using pointer_t = typename super::address_space_fundamental::type;
297 using reference_t = typename std::remove_pointer_t<pointer_t>&;
298
299 /** Allow initialization of a pointer type from the address of an
300 element with the same type and address space
301 */
302 address_space_ptr(address_space_fundamental<typename std::pointer_traits<T>::element_type, AS> *p)
303 : address_space_fundamental<T, AS> { p->get_address() } {}
304
305 /// Put back the default constructors canceled by the previous definition
306 address_space_ptr() = default;
307};
