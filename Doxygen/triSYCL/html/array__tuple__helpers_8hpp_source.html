26/** Helper to construct an array from initializer elements provided as a
27 tuple
28
29 The trick is to get the std::index_sequence<> that represent 0,
30 1,..., dimension-1 as a variadic template pack Is that we can
31 iterate on, in this function.
32*/
33template <typename V, typename Tuple, std::size_t... Is>
34std::array<typename V::element_type, V::dimension>
35tuple_to_array_iterate(Tuple t, std::index_sequence<Is...>) {
36 /* The effect is like a static for-loop with Is counting from 0 to
37 dimension-1 and thus constructing a uniform initialization { }
38 construction from each tuple element:
39 { std::get<0>(t), std::get<1>(t), ..., std::get<dimension-1>(t) }
40
41 The static cast is here to avoid the warning when there is a loss
42 of precision, for example when initializing an int from a float.
43 */
44 return { { static_cast<typename V::element_type>(std::get<Is>(t))... } };
45}

48/** Construct an array from initializer elements provided as a tuple
49 */
50template <typename V, typename Tuple>
51auto tuple_to_array(Tuple t) {
52 /* Construct an index_sequence with 0, 1, ..., (size of the tuple-1)
53 so that tuple_to_array_iterate can statically iterate on it */
54 return tuple_to_array_iterate<V>(
55 t, std::make_index_sequence<std::tuple_size_v<Tuple>> {});
56}

58/** Allows optional expansion of a 1-element tuple to a V::dimension
59 tuple to replicate scalar values in vector initialization
60*/
61template <typename V, typename Tuple, bool expansion = false>
62struct expand_to_vector {
63 static_assert(V::dimension == std::tuple_size_v<Tuple>,
64 "The number of elements in initialization should match the "
65 "dimension of the vector");
66
67 // By default, act as a pass-through and do not do any expansion
68 static auto expand(Tuple t) { return t; }
69};
70
71/** Specialization in the case we ask for expansion */
72template <typename V, typename Tuple> struct expand_to_vector<V, Tuple, true> {
73 static_assert(std::tuple_size_v<Tuple> == 1,
74 "Since it is a vector initialization from a scalar there "
75 "should be only one initializer value");
76
77 /** Construct a tuple from a value
78
79 \param value is used to initialize each tuple element
80
81 \param size is the number of elements of the tuple to be generated
82
83 The trick is to get the std::index_sequence<> that represent 0,
84 1,..., dimension-1 as a variadic template pack Is that we can
85 iterate on, in this function.
86 */
87 template <typename Value, std::size_t... Is>
88 static auto fill_tuple(Value e, std::index_sequence<Is...>) {
89 /* The effect is like a static for-loop with Is counting from 0 to
90 dimension-1 and thus replicating the pattern to have
91 make_tuple( (0, e), (1, e), ... (n - 1, e) )
92
93 Since the "," operator is just here to throw away the Is value
94 (which is needed for the pack expansion...), at the end this is
95 equivalent to:
96 make_tuple( e, e, ..., e )
97 */
98 return std::make_tuple(((void)Is, e)...);
99 }
100
101
102 /** We expand the 1-element tuple by replicating into a tuple with the
103 size of the vector */
104 static auto expand(Tuple t) {
105 return fill_tuple(std::get<0>(t),
106 std::make_index_sequence<V::dimension>{});
107 }
108};

110/** Create the array data of V from a tuple of initializer
111
112 If there is only 1 initializer, this is a scalar initialization of a
113 vector and the value is expanded to all the vector elements first.
114*/
115template <typename V, typename Tuple>
116auto expand(Tuple t) {
117 return tuple_to_array<V>(expand_to_vector<V,
118 decltype(t),
119 /* Only ask the expansion to all vector
120 element if there only a scalar
121 initializer */
122 std::tuple_size_v<Tuple> == 1>{}.expand(t));
123}
