19/** A simple thread safe cache mechanism to cache std::shared_ptr of
20 values indexed by keys
21
22 Since internally only std::weak_ptr are stored, this does not
23 prevent object deletion but it is up to the programmer not to use
24 this cache to retrieve deleted objects.
25*/
26template <typename Key, typename Value>
27class cache {
28
29public:
30
31 /// The type of the keys used to indexed the cache
32 using key_type = Key;
33
34 /// The base type of the values stored in the cache
35 using value_type = Value;
36
37private:
38
39 /// The caching storage
40 std::unordered_map<key_type, std::weak_ptr<value_type>> c;
41
42 /// To make the cache thread-safe
43 std::mutex m;
44
45public:
46
47 /** Get a value stored in the cache if present or insert by calling
48 a generator function
49
50 \param[in] k is the key used to retrieve the value
51
52 \param[in] create_element is the function to be called if the
53 key is not found in the cache to generate a value which is
54 inserted for the key. This function has to produce a value
55 convertible to a shared_ptr
56
57 \return a shared_ptr to the value retrieved or inserted
58 */
59 template <typename Functor>
60 std::shared_ptr<value_type> get_or_register(const key_type &k,
61 Functor &&create_element) {
62 std::lock_guard<std::mutex> lg { m };
63
64 auto i = c.find(k);
65 if (i != c.end())
66 if (auto observe = i->second.lock())
67 // Returns \c shared_ptr only if target object is still alive
68 return observe;
69
70 // Otherwise create and insert a new element
71 std::shared_ptr<value_type> e { create_element() };
72 c.insert({ k, e });
73 return e;
74 }
75
76
77 /** Remove an entry from the cache
78
79 \param[in] k is the key associated to the value to remove from
80 the cache
81 */
82 void remove(const key_type &k) {
83 std::lock_guard<std::mutex> lg { m };
84 c.erase(k);
85 }
86
87};
88
89}
