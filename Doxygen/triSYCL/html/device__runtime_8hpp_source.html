47/// SYCL accessor seen from a device perspective
48template <typename Accessor>
49class accessor :
50 public detail::container_element_aspect<typename Accessor::value_type> {
51
52 /** The pointer to the data
53
54 For now none of the following works yet for various reasons:
55 \code
56 trisycl::global_ptr<typename Accessor::value_type> buffer;
57 trisycl::global<typename Accessor::value_type *> buffer;
58 __global typename Accessor::pointer buffer;
59 \endcode
60
61 So use a more trivial approach
62 */
63 TRISYCL_GLOBAL_AS typename Accessor::value_type *buffer;
64
65 /** The size of the accessor
66
67 The compiler should optimize it out if it is unused */
68 std::size_t size;
69
70public:
71
72 /// Capture the minimal accessor features
73 accessor(Accessor &a) :
74 /* Initialize the buffer with actually a \c cl_mem since it will be
75 overridden with the kernel argument by the serializer at
76 run-time, but at least from the kernel point of view it has to
77 look like an \c Accessor::value_type * for the kernel
78 outlining */
79 buffer {
80 reinterpret_cast<TRISYCL_GLOBAL_AS typename Accessor::value_type *>
81 (a.implementation->get_order())
82 },
83 size { a.get_count() }
84 {
85#ifndef TRISYCL_DEVICE
86 /* Register the buffer address to be updated with the final
87 version before the kernel set arg */
88 TRISYCL_DUMP_T("drt::accessor(Accessor &a) : &a = "
89 << (void *) &a << '\n'
90 << "\t&buffer = " << (void *) &buffer
91 << ", Order = " << a.implementation->get_order());
92#endif
93 }
94
95
96 /// use the accessor with integers à la []
97 auto &operator[](std::size_t index) const noexcept {
98 return buffer[index];
99 }
100
101};
