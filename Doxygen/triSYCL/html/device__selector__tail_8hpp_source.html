25inline device device_selector::select_device(vector_class<platform> platforms) const {
26 int best_id = -1;
27 int best_score = -1;
28 device best_device;
29
30 for (auto &platform : platforms) {
31 for (auto &dev : platform.get_devices()) {
32 int score = operator()(dev);
33 if (score > best_score) {
34 best_id = 1;
35 best_device = dev;
36 best_score = score;
37 }
38 }
39 }
40 if (best_id == -1)
41 throw ::trisycl::runtime_error("Could not find device");
42
43 return best_device;
44}

54class device_type_selector : public device_selector {
55
56private:
57
58 /// The device_type to select
59 info::device_type device_type;
60
61 /** Cache the default device to select with the default device
62 selector.
63
64 This is the host device at construction time and remains as is
65 if there is no openCL device */
66 device default_device;
67
68public:
69
70 device_type_selector(info::device_type device_type)
71 : device_type { device_type } {
72 // The default device selection heuristic
73#ifdef TRISYCL_OPENCL
74 if (device_type == info::device_type::automatic) {
75 // Ask Boost.Compute for the default OpenCL device
76 try {
77 default_device = boost::compute::system::default_device();
78 }
79 catch (...) {
80 /* If there is no OpenCL device, just keep the
81 default-constructed device, which is the host device */
82 }
83 }
84#endif
85 }
86
87
88 // To select only the requested device_type
89 int operator()(const device &dev) const override {
90 if (device_type == info::device_type::all)
91 // All devices fit all
92 return 1;
93
94 if (device_type == info::device_type::automatic)
95 // Only select the default device
96 return dev == default_device ? 1 : -1;
97
98 if (device_type == info::device_type::opencl)
99 // For now, any non host device is an OpenCL device
100 return dev.is_host() ? -1 : 1;
101
102 return dev.type() == device_type ? 1 : -1;
103 }
104
105};
