24vector_class<device>
25device::get_devices(info::device_type device_type) {
26 // Start with the default device
27 vector_class<device> devices = { {} };
28
29#ifdef TRISYCL_OPENCL
30 // Then add all the OpenCL devices
31 for (const auto &d : boost::compute::system::devices())
32 devices.emplace_back(d);
33#endif
34
35 // The selected devices
36 vector_class<device> sd;
37 device_type_selector s { device_type };
38
39 // Return the devices with the good criterion according to the selector
40 std::copy_if(devices.begin(), devices.end(), std::back_inserter(sd),
41 [&](const device &e) { return s(e) >= 0; });
42 return sd;
43}
