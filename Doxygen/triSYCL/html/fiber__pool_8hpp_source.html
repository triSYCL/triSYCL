4/** \file

6 A Boost.Fiber pool implemented on a std::thread pool

8 The use case is for circuit emulation when there are a lot of
9 fibers launched at the beginning and they have to run concurrently.

85 /// Create a fiber_pool
86 fiber_pool(int thread_number,
87 sched scheduler,
88 bool suspend)
89 : starting_block { static_cast<unsigned int>(thread_number) + 1 }
90 , finish_line { static_cast<unsigned int>(thread_number) }
91 , s { scheduler }
92 {
93 if (scheduler == sched::shared_work)
94 // This scheduler needs a shared context
95 pc_shared = pooled_shared_work::create_pool_ctx(suspend);
96 else if (scheduler == sched::work_stealing)
97 // This scheduler needs a shared context
98 pc_stealing = pooled_work_stealing::create_pool_ctx
99 (thread_number, suspend);
100 // Start the working threads
101 working_threads = ranges::iota_view { 0, thread_number }
102 | ranges::views::transform([&] (int i) {
103 return std::async(std::launch::async,
104 [&, i] { run(i); }); })
105 | ranges::to<std::vector>;
106 // Wait for all thread workers to be ready
107 starting_block.count_down_and_wait();
108 }

111 /** Submit some work on a new fiber
112
113 \param[in] work is the callable to execute, taking no arguments
114 and returning a result of some type R
115
116 \return a future<R>
117 */
118 template <typename Callable>
119 auto submit(Callable && work) {
120 // Put the packaged_task into a shared_ptr to fit it later in a
121 // lambda in a std::function since the type is move-only
122 auto pt = std::make_shared<boost::fibers::packaged_task<void(void)>>
123 ([w = std::forward<Callable>(work)] () mutable { return w(); });
124 auto f = pt->get_future();
125 // Submit a lambda to do the type erasure so the submission queue
126 // type is independent of the type of the packaged_task
127 submission.push([p = std::move(pt)] () mutable { (*p)(); });
128 // Return the future to the client to get the result or the exception
129 return f;
130 }
131
132
133 /// Close the submission
134 void close() {
135 // Can be done many times, so no protection required here
136 submission.close();
137 }
138
139
140 /// Wait for all the threads are done
141 void join() {
142 // Can be done only once
143 if (joinable) {
144 // Close the submission if not done already
145 close();
146 for (auto &t : working_threads)
147 // A Boost.Fiber scheduler will block its thread if they still
148 // have some work to do
149 t.get();
150 joinable = false;
151 }
152 }
153
154
155 /// Wait for some remaining work to be done
156 ~fiber_pool() {
157 // Join first if not done already
158 join();
159 // There is a circular dependecy of shared_ptr/intrusive_ptr between
160 // pc_stealing and its schedulers_. after we join we know none of the
161 // schedulers_ are still running so we break the cycle and everything
162 // cleansup itself.
163 if (pc_stealing)
164 pc_stealing->schedulers_.clear();
165 }
166
167private:
168
169 /// The thread worker job
170 void run(int i) {
171 if (s == sched::shared_work)
172 boost::fibers::use_scheduling_algorithm<pooled_shared_work>(pc_shared);
173 else if (s == sched::work_stealing)
174 boost::fibers::use_scheduling_algorithm
175 <pooled_work_stealing>(pc_stealing);
176 // Otherwise a round-robin scheduler is used and the fibers will
177 // use only 1 thread since there is no thread migration in that
178 // case
179
180 // Wait for all thread workers to be ready
181 starting_block.count_down_and_wait();
182
183 // Only the first thread receives and starts the work
184 if (i == 0) {
185 // Keep track of each fiber execution to avoid quitting before completion
186 std::vector<boost::fibers::fiber> fibers;
187 for (;;) {
188 decltype(submission)::value_type work;
189 if (submission.pop(work)
190 == boost::fibers::channel_op_status::closed)
191 // Someone asked to stop accepting work
192 break;
193 // Launch the work on a new fiber
194 fibers.emplace_back(starting_mode, std::move(work));
195 }
196 // Now wait for the completion of each fiber
197 for (auto &f : fibers)
198 f.join();
199 }
200 // Wait for all the threads to finish their fiber execution
201 finish_line.wait();
202 }
203
204};
