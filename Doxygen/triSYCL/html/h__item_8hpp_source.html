28/** \c h_item<int dimensions> identifies an instance of a \c
29 group::parallel_for_work_item function object executing at each
30 point in a local \c range<int dimensions> passed to a \c
31 parallel_for_work_item call or to the corresponding \c
32 parallel_for_work_group call if no range is passed to the \c
33 parallel_for_work_item call. It encapsulates enough information to
34 identify the work-item’s local and global items according to the
35 information given to \c parallel_for_work_group (physical ids) as
36 well as the work-item’s logical local items in the flexible
37 range. All returned items objects are offset-less. Instances of
38 the \c h_item<int dimensions> class are not user-constructible and
39 are passed by the runtime to each instance of the function object.
40*/
41template <int Dimensions = 1>
42struct h_item : boost::equality_comparable<h_item<Dimensions>> {

208 */

211#if defined(_OPENMP) && !defined(TRISYCL_NO_BARRIER)
212 /* Use OpenMP barrier in the implementation with 1 OpenMP thread per
213 work-item of the work-group */
214#pragma omp barrier
215#else
216 // \todo To be implemented efficiently otherwise
217 TRISYCL_UNIMPL;
218#endif
219 }
