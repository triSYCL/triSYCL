163 /** Schedule the kernel
164
165 Add a traced version of the kernel in host mode or add the
166 kernel in an instantiating function for later extraction by the
167 compiler
168 */
169 template <typename KernelName,
170 typename Kernel>
171 void schedule_kernel(Kernel k) {
172 /* Explicitly capture task by copy instead of having this captured
173 by reference and task by reference by side effect */
174 task->schedule(detail::trace_kernel<KernelName>([=, t = task] () mutable {
175 if (t->owner_queue->is_host())
176 k();
177 else {
178 TRISYCL_DUMP_T("schedule_kernel &k = " << (void *) &k);
179
180 // Structure the kernel code so it can be outlined and called
181 detail::launch_device_kernel<KernelName>(*t, k);
182 // \todo for now only deal with 1 physical work-item only
183 t->get_kernel().single_task(t, t->get_queue());
184 }
185 }));
186 }
