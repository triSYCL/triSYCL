22/** Avoid the interprocedural optimization to remove these arguments

36/** Instantiate the kernel code
37
38 To have a clear view of what a kernel is at the LLVM IR level even
39 when a lot of optimizations are done (-O3), call the kernel from
40 this function marked as \c noinline, otherwise the kernel code is
41 inlined directly where it is used.
42
43 Actually some optimizations/code restructuring might duplicate the
44 call to \c instantiate_kernel but the \c noduplicate attribute
45 seems incompatible with \c noinline in Clang/LLVM 3.9.1, because
46 the inlining happen when both attributes are used...
47*/
48template <typename KernelName,
49 typename Functor>
50[[gnu::noinline]] void
51instantiate_kernel(Functor f) noexcept {
52 /* The outlining compiler is expected to do some massage here or
53 around and to insert some calls to \c serialize_arg and so on */
54 f();
55}

58/** Set the task to be used to launch the kernel
59
60 This is actually a dummy function so that the device compiler can

63
64 It is better to use a weak linkage than having this function with

81/** Gather in a single function what is necessary for the device
82 compiler to outline the kernel or to set up the kernel and
83 serialize the arguments.
84
85 The \c noinline attribute is used to force this to appear as a
86 real function the device compiler can massage.
87*/
88template <typename KernelName,
89 typename Kernel>
90[[gnu::noinline]]
91void launch_device_kernel(detail::task &t,
92 Kernel k) noexcept {
93 /** Setup the task to be used to launch the kernel.
94
95 This is used by the outlining compiler. */
96 detail::set_kernel_task_marker(t);
97 /* A simplified version for the device just to be able to extract
98 the kernel itself.
99
100 It is not a real outlining but a good approximation to be
101 massaged by a compiler later.
102
103 Put the iteration space loops in the kernels for now, which makes
104 sense for CPU emulation or FPGA pipelined execution.
105 */
106 instantiate_kernel<KernelName>(k);
107}
