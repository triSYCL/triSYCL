24/** A SYCL item stores information on a work-item with some more context
25 such as the definition range and offset.
26*/
27template <int Dimensions = 1, bool with_offset = true>
28class item : boost::equality_comparable<item<Dimensions>> {
29
30public:
31 /// The number of dimensions of the item
32 static auto constexpr rank() { return Dimensions; }
33
34private:
35
36 range<Dimensions> global_range;
37 id<Dimensions> global_index;
38 id<Dimensions> offset;
39
40public:
41
42 /** Create an item from a local size and an optional offset
43
44 This constructor is used by the triSYCL implementation and the
45 non-regression testing.
46 */
47 item(range<Dimensions> global_size,
48 id<Dimensions> global_index,
49 id<Dimensions> offset = {}) :
50 global_range { global_size },
51 global_index { global_index },
52 offset { offset }
53 {}
54
55
56 /** To be able to copy and assign item, use default constructors too
57
58 \todo Make most of them protected, reserved to implementation
59 */
60 item() = default;
61
62
63 /** Return the constituent local or global id<> representing the
64 work-item's position in the iteration space
65 */
66 id<Dimensions> get_id() const { return global_index; }
67
68
69 /** Return the requested dimension of the constituent id<> representing
70 the work-item's position in the iteration space
71 */
72 size_t get_id(int dimension) const { return get_id()[dimension]; }
73
74
75 /** Return the constituent id<> l-value representing the work-item's
76 position in the iteration space in the given dimension
77 */
78 auto &operator[](int dimension) { return global_index[dimension]; }
79
80
81 /** Returns a range<> representing the dimensions of the range of
82 possible values of the item
83 */
84 range<Dimensions> get_range() const { return global_range; }
85
86 /// Return the same value as get_range().get(dimension)
87 size_t get_range(int dimension) const { return get_range().get(dimension); }
88
89 /** Returns an id<> representing the n-dimensional offset provided to
90 the parallel_for and that is added by the runtime to the global-ID
91 of each work-item, if this item represents a global range
92
93 For an item representing a local range of where no offset was passed
94 this will always return an id of all 0 values.
95 */
96 id<Dimensions> get_offset() const {
97 static_assert(with_offset == true, "get_offset only callable with has_offset = true");
98 return offset;
99 }
100
101
102 /** Return the linearized ID in the item's range
103
104 Computed as the flatted ID after the offset is subtracted.
105 */
106 size_t get_linear_id() const {
107 return detail::linear_id(get_range(), get_id(), get_offset());
108 }
109
110
111 /** For the implementation, need to set the global index
112
113 \todo Move to private and add friends
114 */
115 void set(id<Dimensions> Index) { global_index = Index; }
116
117 /** Returns an item with same dimensions but offset set to 0 */
118 operator item<Dimensions, true> () const {
119 static_assert(with_offset == false, "get_offset only callable with has_offset = true");
120 return { this->get_range(), this->get_id(), 0 };
121 }
122
123 /// Display the value for debugging and validation purpose
124 void display() const {
125 global_range.display();
126 global_index.display();
127 offset.display();
128 }
129
130 /// Comparison operators for item
131 bool operator==(const item<Dimensions> &itemB) const {
132 return (global_range == itemB.global_range &&
133 global_index == itemB.global_index &&
134 offset == itemB.offset);
135 }
136};
