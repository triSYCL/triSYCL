20/** Compute a linearized array access used in the OpenCL 2 world
21
22 Typically for the get_global_linear_id() and get_local_linear_id()
23 functions.

25template <typename Range, typename Id>
26size_t constexpr inline linear_id(Range range, Id id, Id offset = {}) {
27 auto dims = std::distance(std::begin(range), std::end(range));
28
29 size_t linear_id = 0;
30 /* A good compiler should unroll this and do partial evaluation to
31 remove the first multiplication by 0 of this Horner evaluation and
32 remove the 0 offset evaluation */
33 for (int i = dims - 1; i >= 0; --i)
34 linear_id = linear_id*range[i] + id[i] - offset[i];
35
36 return linear_id;
37}
