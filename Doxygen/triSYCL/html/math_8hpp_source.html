29/// Declare a FUN unary function for scalar and vector types
30#define TRISYCL_MATH_WRAP(FUN) template<typename T> \
31 /* Just forward the scalar case to the C++ math library */ \
32 T FUN(const T& x) { \
33 return std::FUN(x); \
34 } \
35 /* Declare the vector version as applying the scalar version on each \
36 element */ \
37 template <typename T, int size> \
38 auto FUN(const vec<T, size>& x) { \
39 return x.map(FUN<T>); \
40 }

299// Returns the cross product of x.xyz and y.xyz. The w component of float4
300// result returned will be 0.0.
301template <typename T, int size>
302auto cross(const vec<T, size>& x, const vec<T, size>& y) {
303 static_assert((size == 4 || size == 3),
304 "vec type should be 3 or 4 dimensions");
305 if constexpr (size > 3) {
306 return vec<T, size> {
307 (x[1] * y[2]) - (x[2] * y[1]),
308 (x[2] * y[0]) - (x[0] * y[2]),
309 (x[0] * y[1]) - (x[1] * y[0]),
310 0.0};
311 } else {
312 return vec<T, size> {
313 (x[1] * y[2]) - (x[2] * y[1]),
314 (x[2] * y[0]) - (x[0] * y[2]),
315 (x[0] * y[1]) - (x[1] * y[0])};
316 }
317}

319// Return the length of vector x, i.e., sqrt(x.x^2 + x.y^2 + ...)
320template <typename T, int size>
321auto length(const vec<T, size>& x) {
322 auto temp = x * x;
323 return sqrt(std::accumulate(temp.begin(), temp.end(), T{0}));
324}

338// returns x if x is +/- 0,

342template<typename T>
343T sign(const T & val) {
344 if (val == 0) return val;
345 if (isnan(val)) return 0;
