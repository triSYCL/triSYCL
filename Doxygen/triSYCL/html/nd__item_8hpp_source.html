27/** A SYCL nd_item stores information on a work-item within a work-group,
28 with some more context such as the definition ranges.

30template <int Dimensions = 1>
31struct nd_item : boost::equality_comparable<nd_item<Dimensions>> {

45 /** Create an empty nd_item<> from an nd_range<>
46
47 \todo This is for the triSYCL implementation which is expected to

51 nd_item(nd_range<Dimensions> ndr) : ND_range { ndr } {}

54 /** Create a full nd_item
55
56 \todo This is for validation purpose. Hide this to the programmer
57 somehow
58 */
59 nd_item(id<Dimensions> global_index,

61 global_index { global_index },

64 ndr.get_local_range() } },

195 The current work-item will wait at the barrier until all work-items

197
198 In addition, the barrier performs a fence operation ensuring that all
199 memory accesses in the specified address space issued before the
200 barrier complete before those issued after the barrier
201 */

203 access::fence_space::global_and_local) const {
204#if defined(_OPENMP) && !defined(TRISYCL_NO_BARRIER)
205 /* Use OpenMP barrier in the implementation with 1 OpenMP thread per
206 work-item of the work-group */
207#pragma omp barrier
208#else
209 // \todo To be implemented efficiently otherwise
210 TRISYCL_UNIMPL;
211#endif
212 }
