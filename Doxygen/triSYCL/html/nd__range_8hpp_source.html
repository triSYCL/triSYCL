23/** A ND-range, made by a global and local range, to specify work-group
24 and work-item organization.
25
26 The local offset is used to translate the iteration space origin if
27 needed.
28
29 \todo add copy constructors in the specification
30*/
31template <int Dimensions = 1>
32struct nd_range : boost::equality_comparable<nd_range<Dimensions>> {
33 /// The number of dimensions of the nd_range
34 static auto constexpr rank() { return Dimensions; }
35
36private:
37
38 range<rank()> global_range;
39 range<rank()> local_range;
40 id<rank()> offset;
41
42public:
43
44 /** Construct a ND-range with all the details available in OpenCL
45
46 By default use a zero offset, that is iterations start at 0
47 */
48 nd_range(range<Dimensions> global_size,
49 range<Dimensions> local_size,
50 id<Dimensions> offset = {}) :
51 global_range { global_size }, local_range { local_size }, offset { offset }
52 { }
53
54
55 /// Get the global iteration space range
56 range<Dimensions> get_global_range() const { return global_range; }
57
58
59 /// Get the local part of the iteration space range
60 range<Dimensions> get_local_range() const { return local_range; }
61
62
63 /// Get the range of work-groups needed to run this ND-range
64 auto get_group_range() const {
65 /* This is basically global_range/local_range, round up to the
66 next integer, in case the global range is not a multiple of the
67 local range. Note this is a motivating example to build a range
68 from a scalar with a broadcasting constructor. */
69 return (global_range + local_range - range<Dimensions>{ 1 })/local_range;
70 }
71
72
73 /// \todo get_offset() is lacking in the specification
74 id<Dimensions> get_offset() const { return offset; }
75
76
77 /// Display the value for debugging and validation purpose
78 void display() const {
79 global_range.display();
80 local_range.display();
81 offset.display();
82 }
83
84 /// Comparison operators for nd_range.
85 bool operator==(const nd_range &nd_rangeB) const {
86 return (global_range == nd_rangeB.global_range &&
87 offset == nd_rangeB.offset);
88 }
89};
