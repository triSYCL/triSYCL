20/// Some implementation details about the SYCL queue

26 /** A cache to always return the same alive queue for a given OpenCL

69 /// Get a singleton instance of the opencl_queue
70 static std::shared_ptr<opencl_queue>

74 }

76
77 /** Create a new queue associated to this device

85 return instance (boost::compute::command_queue {

88 d.get_boost_compute()
89 });
90 }
