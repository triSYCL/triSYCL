77/* When passing the arguments to Boost Compute, we need to know if we have to
78 unwrap our type and give Boost Compute the actual OpenCL type. To do this
79 we define this meta-function that returns true if \c T is a wrapper type
80 to an OpenCL type. For example \c is_wrapper<trisycl::int3>::value == true
81*/
82template<class T>
83struct is_wrapper : std::false_type {};

85#define TRISYCL_IS_WRAPPER_TRAIT(type) \
86template <> struct is_wrapper<type> : std::true_type {};

88/* We define 3 different wrapper classes around OpenCL types.
89 These classes allow us to use t.x() and t.y() for vector types of size 2.
90 And in addition we can also use t.z() and t.w() for vector types of size
91 3 and 4 similarly to OpenCL vector types
92*/
93#define TRISYCL_WRAPPER_CLASS_2(cl_type, boost_name, scalar_type) \
94 class cl_type { \
95 ::cl_type self; \
96 \
97 public: \
98 cl_type () = default; \
99 cl_type (::cl_type self_) : self { self_ } {} \
100 cl_type (scalar_type x, scalar_type y) : self { x, y } {} \
101 auto& x() { return self.s[0]; } \
102 auto& y() { return self.s[1]; } \
103 auto& unwrap() const { return self; }}; \
104 TRISYCL_IS_WRAPPER_TRAIT(cl_type)
105
106#define TRISYCL_WRAPPER_CLASS_3(cl_type, boost_name, scalar_type) \
107 class cl_type { \
108 ::cl_type self; \
109 \
110 public: \
111 cl_type () = default; \
112 cl_type (::cl_type self_) : self { self_ } {} \
113 cl_type (scalar_type x, scalar_type y, scalar_type z) : \
114 self { x, y, z } {} \
115 auto& x() { return self.s[0]; } \
116 auto& y() { return self.s[1]; } \
117 auto& z() { return self.s[2]; } \
118 auto& unwrap() const { return self; }}; \
119 TRISYCL_IS_WRAPPER_TRAIT(cl_type)
120
121#define TRISYCL_WRAPPER_CLASS_4(cl_type, boost_name, scalar_type) \
122 class cl_type { \
123 ::cl_type self; \
124 \
125 public: \
126 cl_type () = default; \
127 cl_type (::cl_type self_) : self { self_ } {} \
128 cl_type (scalar_type x, scalar_type y, scalar_type z, scalar_type w) : \
129 self { x, y, z, w } {} \
130 auto& x() { return self.s[0]; } \
131 auto& y() { return self.s[1]; } \
132 auto& z() { return self.s[2]; } \
133 auto& w() { return self.s[3]; } \
134 auto& unwrap() const { return self; }}; \
135 TRISYCL_IS_WRAPPER_TRAIT(cl_type)

148// Helper macro to properly define each OpenCL type
149#define TRISYCL_H_DEFINE_TYPE(cl_type, boost_name, scalar_type, i) \
150 BOOST_PP_IF(BOOST_PP_EQUAL(i, 1), TRISYCL_TYPEDEF_TYPE, \

156 TRISYCL_TYPEDEF_TYPE)))) \
