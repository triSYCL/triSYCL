69template <typename Range, typename ParallelForFunctor>
70void parallel_for_iterate(Range r, ParallelForFunctor &f)
71{
72 tbb::parallel_for(to_tbb_range(r),
73 [&](auto idx) { f(to_sycl_id(idx)); },
74 tbb::simple_partitioner{});
75}

89template <int Dimensions = 1, typename ParallelForFunctor>

102/** Calls the appropriate ternary parallel_for overload based on the
103 index type of the kernel function object f
104*/
105template <int Dimensions = 1, typename ParallelForFunctor>
106void parallel_for(range<Dimensions> r, ParallelForFunctor f)
107{
108 using mf_t = decltype(std::mem_fn(&ParallelForFunctor::operator()));
109 using arg_t = typename mf_t::second_argument_type;
110 parallel_for(r, f, arg_t{});
111}

127/// Implement the loop on the work-groups
128template <int Dimensions = 1, typename ParallelForFunctor>
129void parallel_for_workgroup(nd_range<Dimensions> r, ParallelForFunctor f)
130{
131 auto reconstruct_group = [&](id<Dimensions> l) {
132 group<Dimensions> group{l, r};
133 f(group);
134 };
135
136 parallel_for_iterate(r.get_group_range(), reconstruct_group);

139/// Implement the loop on the work-items inside a work-group
140template <int Dimensions, typename T_Item, typename ParallelForFunctor>
141void parallel_for_workitem(const group<Dimensions> &g,
142 ParallelForFunctor f)
143{
144 auto reconstruct_item = [&](id<Dimensions> local) {
145 T_Item index{g.get_nd_range()};
146 index.set_local(local);
147 index.set_global(local +
148 id<Dimensions>(g.get_local_range()) * g.get_id());
149 f(index);
150 };
151
152 parallel_for_iterate(g.get_local_range(), reconstruct_item);
153}

155/// Implement a variation of parallel_for to take into account a nd_range<>
156template <int Dimensions = 1, typename ParallelForFunctor>
157void parallel_for(nd_range<Dimensions> r, ParallelForFunctor f)
158{
159 auto iterate_in_work_group = [&](id<Dimensions> g) {
160 trisycl::group<Dimensions> wg{g, r};
161 parallel_for_workitem<Dimensions, nd_item<Dimensions>, decltype(f)>(
162 wg, f);
163 };
164
165 parallel_for_iterate(r.get_group_range(), iterate_in_work_group);
166}
