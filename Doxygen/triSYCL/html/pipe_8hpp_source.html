49 /** Finds the smallest integral power of two not less than the given
50 value but at least 2 to fit the requirement from
51 boost::fibers::buffered_channel
52
53 \todo At some point use C++20 std::bit_ceil instead of this
54 DIY function */
55 static constexpr std::size_t bit_ceil_at_least_2(std::size_t value) {
56 std::size_t result = 2;
57 while (value > result)
58 result *= 2;
59 return result;
60 }
61
62public:
63
64 // The data type transmitted by the pipe
65 using value_type = T;
66
67 /** The minimum capacity in value_type of this pipe.
68
69 The underlying implementation requires a power of 2 */
70 static constexpr SizeT min_capacity = bit_ceil_at_least_2(MinCapacity);
71
72 // The data type used to represent the capacity itself
73 using size_type = SizeT;
74
75private:
76
77 /// A FIFO used to implement the pipe
78 static inline boost::fibers::buffered_channel<value_type>
79 channel { min_capacity };
80
81public:
82
83 /** Blocking read from the pipe
84
85 \return the value
86 */
87 static value_type read() {
88 return channel.value_pop();
89 }
90
91
92 /** Blocking write to the pipe
93
94 \param[in] value is the data to send to the pipe
95 */
96 static void write(const value_type &value) {
97 channel.push(value);
98 }
99
100
101 /** Non-blocking read from the pipe
102
103 \param[out] success_code is true iff the read was successful
104
105 \return the value read from the pipe or the default one
106 */
107 static value_type read(bool &success_code) {
108 value_type value;
109 success_code =
110 channel.try_pop(value) == boost::fibers::channel_op_status::success;
111 /* The current API might be inefficient if we copy always value back.
112 But we should have a copy elision... */
113 return value;
114 }
115
116
117 /** Non-blocking write to the pipe
118
119 \param[out] success_code is true iff the read was successful
120
121 \return the value read from the pipe or the default one
122 */
123 static void write(const value_type &data, bool &success_code) {
124 success_code =
125 channel.try_push(data) != boost::fibers::channel_op_status::full;
126 }
