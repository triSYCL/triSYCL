46template <typename T,
47 access::mode AccessMode,
48 access::target Target>
49class pipe_accessor :
50 public detail::debug<detail::sycl_2_2::pipe_accessor<T, AccessMode, Target>>
51{
52public:
53
54 static constexpr auto rank = 1;
55 static constexpr auto mode = AccessMode;
56 static constexpr auto target = Target;
57
58 static constexpr bool blocking =
59 (target == trisycl::access::target::blocking_pipe);

61 /// The STL-like types
62 using value_type = T;
63 using reference = value_type&;
64 using const_reference = const value_type&;

66private:
67
68 /// The real pipe implementation behind the hood
69 std::shared_ptr<detail::sycl_2_2::pipe<T>> implementation;
70
71 /** Store the success status of last pipe operation
72
73 It is not impacted by reservation success.
74
75 It does exist even if the pipe accessor is not evaluated in a
76 boolean context for, but a use-def analysis can optimise it out
77 in that case and not use some storage
78
79 Use a mutable state here so that it can work with a [=] lambda
80 capture without having to declare the whole lambda as mutable
81 */
82 bool mutable ok = false;
83
84public:

86 /** Construct a pipe accessor from an existing pipe
87 */
88 pipe_accessor(const std::shared_ptr<detail::sycl_2_2::pipe<T>> &p)
89 : implementation { p } {
90 // TRISYCL_DUMP_T("Create a kernel pipe accessor write = "
91 // << is_write_access());
92 // Verify that the pipe is not already used in the requested mode
93 if (mode == access::mode::write)
94 if (implementation->used_for_writing)
95 /// \todo Use pipe_exception instead
96 throw std::logic_error { "The pipe is already used for writing." };
97 else
98 implementation->used_for_writing = true;
99 else
100 if (implementation->used_for_reading)
101 throw std::logic_error { "The pipe is already used for reading." };
102 else
103 implementation->used_for_reading = true;
104 }

107 /** Construct a pipe accessor from an existing pipe
108
109 For now the handler is not used.
110 */
111 pipe_accessor(const std::shared_ptr<detail::sycl_2_2::pipe<T>> &p,
112 handler &command_group_handler)
113 : pipe_accessor(p) {}

163 /** In an explicit bool context, the accessor gives the success
164 status of the last access
165
166 It is not impacted by reservation success.
167
168 The explicitness is related to avoid \code some_pipe <<
169 some_value \endcode to be interpreted as \code some_bool <<
170 some_value \endcode when the type of \code some_value \endcode
171 is not the same type as the pipe type.
172
173 \return true on success of the previous read or write operation
174 */
175 explicit operator bool() const {
176 return ok;
177 }
178
179
180 /** Try to write a value to the pipe
181
182 \param[in] value is what we want to write
183
184 \return this so we can apply a sequence of write for example
185 (but do not do this on a non blocking pipe...)
186
187 \todo provide a && version
188
189 This function is const so it can work when the accessor is
190 passed by copy in the [=] kernel lambda, which is not mutable by
191 default
192 */
193 const pipe_accessor &write(const value_type &value) const {
194 static_assert(mode == access::mode::write,
195 "'.write(const value_type &value)' method on a pipe accessor"
196 " is only possible with write access mode");
197 ok = implementation->write(value, blocking);
198 // Return a reference to *this so we can apply a sequence of write
199 return *this;
200 }
201
202
203 /** Some syntactic sugar to use \code a << v \endcode instead of
204 \code a.write(v) \endcode */
205 const pipe_accessor &operator<<(const value_type &value) const {
206 static_assert(mode == access::mode::write,
207 "'<<' operator on a pipe accessor is only possible"
208 " with write access mode");
209 // Return a reference to *this so we can apply a sequence of >>
210 return write(value);
211 }
212
213
214 /** Try to read a value from the pipe
215
216 \param[out] value is the reference to where to store what is
217 read
218
219 \return \code this \endcode so we can apply a sequence of read
220 for example (but do not do this on a non blocking pipe...)
221
222 This function is const so it can work when the accessor is
223 passed by copy in the [=] kernel lambda, which is not mutable by
224 default
225 */
226 const pipe_accessor &read(value_type &value) const {
227 static_assert(mode == access::mode::read,
228 "'.read(value_type &value)' method on a pipe accessor"
229 " is only possible with read access mode");
230 ok = implementation->read(value, blocking);
231 // Return a reference to *this so we can apply a sequence of read
232 return *this;
233 }
234
235
236 /** Read a value from a blocking pipe
237
238 \return the read value directly, since it cannot fail on
239 blocking pipe
240
241 This function is const so it can work when the accessor is
242 passed by copy in the [=] kernel lambda, which is not mutable by
243 default
244 */
245 value_type read() const {
246 static_assert(mode == access::mode::read,
247 "'.read()' method on a pipe accessor is only possible"
248 " with read access mode");
249 static_assert(blocking,
250 "'.read()' method on a pipe accessor is only possible"
251 " with a blocking pipe");
252 value_type value;
253 implementation->read(value, blocking);
254 return value;
255 }
256
257
258 /** Some syntactic sugar to use \code a >> v \endcode instead of
259 \code a.read(v) \endcode */
260 const pipe_accessor &operator>>(value_type &value) const {
261 static_assert(mode == access::mode::read,
262 "'>>' operator on a pipe accessor is only possible"
263 " with read access mode");
264 // Return a reference to *this so we can apply a sequence of >>
265 return read(value);
266 }

272 }

286 ~pipe_accessor() {
287 /// Free the pipe for a future usage for the current mode
288 if (mode == access::mode::write)
289 implementation->used_for_writing = false;
290 else
291 implementation->used_for_reading = false;
292 }
