34/// The implementation of the pipe reservation station
35template <typename PipeAccessor>
36class pipe_reservation :
37 public detail::debug<detail::sycl_2_2::pipe_reservation<PipeAccessor>> {
38 using accessor_type = PipeAccessor;
39 static constexpr bool blocking =
40 (accessor_type::target == trisycl::access::target::blocking_pipe);
41 using value_type = typename accessor_type::value_type;
42 using reference = typename accessor_type::reference;
43
44public:
45
46 using iterator = typename
47 detail::sycl_2_2::pipe<value_type>::implementation_t::iterator;
48 using const_iterator = typename
49 detail::sycl_2_2::pipe<value_type>::implementation_t::const_iterator;
50
51 // \todo Add to the specification
52 static constexpr access::mode mode = accessor_type::mode;
53 static constexpr access::target target = accessor_type::target;
54
55 /** True if the reservation was successful and still uncommitted. B
56 default a pipe_reservation is not reserved and cannot be
57 committed */
58 bool ok = false;
59
60 /// Point into the reservation buffer. Only valid if ok is true
61 typename detail::sycl_2_2::pipe<value_type>::rid_iterator rid;
62
63 /** Keep a reference on the pipe to access to the data and methods
64
65 Note that with inlining and CSE it should not use more register
66 when compiler optimization is in use. */
67 detail::sycl_2_2::pipe<value_type> &p;
68
69
70 /** Test that the reservation is in a usable state
71
72 \todo Throw exception instead
73 */
74 void assume_validity() {
75 assert(ok);
76 }
77
78public:
79
80 /// Create a pipe reservation station that reserves the pipe itself
81 pipe_reservation(detail::sycl_2_2::pipe<value_type> &p,
82 std::size_t s)
83 : p { p } {
84 static_assert(mode == access::mode::write
85 || mode == access::mode::read,
86 "A pipe can only be accesed in read or write mode,"
87 " exclusively");
88
89 /* Since this test is constexpr and dependent of a template
90 parameter, it should be equivalent to a specialization of the
91 method but in a clearer way */
92 if (mode == access::mode::write)
93 ok = p.reserve_write(s, rid, blocking);
94 else
95 ok = p.reserve_read(s, rid, blocking);
96 }
97
98
99 /** No copy constructor with some spurious commit in the destructor
100 of the original object
101 */
102 pipe_reservation(const pipe_reservation &) = delete;
103
104
105 /// Only a move constructor is required to move it into the shared_ptr
106 pipe_reservation(pipe_reservation &&orig) :
107 ok {orig.ok },
108 rid {orig.rid },
109 p { orig.p } {
110 /* Even when an object is moved, the destructor of the old
111 object is eventually called, so leave the old object in a
112 destructable state but without any commit capability */
113 orig.ok = false;
114 }
115
116
117 /** Keep the default constructors too
118
119 Otherwise there is no move semantics and the copy is made by
120 creating a new reservation and destructing the old one with a
121 spurious commit in the meantime...
122 */
123 pipe_reservation() = default;
124
125
126 /** Test if the reservation succeeded and thus if the reservation
127 can be committed
128
129 Note that it is up to the user to ensure that all the
130 reservation elements have been initialized correctly in the case
131 of a write for example
132 */
133 operator bool() {
134 return ok;
135 }
136
137
138 /// Start of the reservation area
139 iterator begin() {
140 assume_validity();
141 return rid->start;
142 }
143
144
145 /// Past the end of the reservation area
146 iterator end() {
147 assume_validity();
148 return rid->start + rid->size;
149 }
150
151
152 /// Get the number of elements in the reservation station
153 std::size_t size() {
154 assume_validity();
155 return rid->size;
156 }
157
158
159 /// Access to an element of the reservation
160 reference operator[](std::size_t index) {
161 assume_validity();
162 TRISYCL_DUMP_T("[] index = " << index
163 << " Reservation write address = " << &(rid->start[index]));
164
165 return rid->start[index];
166 }
167
168
169 /** Commit the reservation station
170
171 \todo Add to the specification that for simplicity a reservation
172 can be commited several times but only the first one is taken
173 into account
174 */
175 void commit() {
176 if (ok) {
177 // If the reservation is in a committable state, commit
178 TRISYCL_DUMP_T("Commit");
179 rid->ready = true;
180 if (mode == access::mode::write)
181 p.move_write_reservation_forward();
182 else
183 p.move_read_reservation_forward();
184 ok = false;
185 }
186 }
187
188
189 /// An implicit commit is made in the destructor
190 ~pipe_reservation() {
191 commit();
192 }
193
194};
