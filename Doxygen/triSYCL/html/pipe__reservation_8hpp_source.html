27/** The pipe reservation station allows to reserve an array-like view
28 inside the pipe for ordered race-free access from various
29 work-items for example
30 */
31template <typename PipeAccessor>
32struct pipe_reservation {
33 using accessor_type = PipeAccessor;
34 static constexpr bool blocking =
35 (accessor_type::target == trisycl::access::target::blocking_pipe);
36 using accessor_detail = typename accessor_type::accessor_detail;
37 /// The STL-like types
38 using value_type = typename accessor_type::value_type;
39 using reference = value_type&;
40 using const_reference = const value_type&;
41 using pointer = value_type*;
42 using const_pointer = const value_type*;
43 using size_type = std::size_t;
44 using difference_type = ptrdiff_t;
45 using iterator =
46 typename detail::sycl_2_2::pipe_reservation<accessor_detail>::iterator;
47 using const_iterator =
48 typename detail::sycl_2_2::pipe_reservation<accessor_detail>::const_iterator;
49 using reverse_iterator = std::reverse_iterator<iterator>;
50 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
51
52 /** Point to the underlying implementation that can be shared in the
53 SYCL model with a handler semantics */
54 typename std::shared_ptr<detail::sycl_2_2::pipe_reservation<accessor_detail>>
55 implementation;
56
57 /** Use default constructors so that we can create a new buffer copy
58 from another one, with either a l-value or a r-value (for
59 std::move() for example).
60
61 Since we just copy the shared_ptr<> above, this is where/how the
62 sharing magic is happening with reference counting in this case.
63 */
64 pipe_reservation() = default;
65
66
67 /// Create a pipe_reservation for an accessor and a number of elements
68 pipe_reservation(accessor_type &accessor, std::size_t s)
69 : implementation {
70 new detail::sycl_2_2::pipe_reservation<accessor_detail> {
71 get_pipe_detail(accessor), s }
72 } {}
73
74
75 /** Create a pipe_reservation from the implementation detail
76
77 This is an internal constructor to allow reserve() on the
78 implementation to lift a full-fledged object through
79 accessor::reserve().
80
81 \todo Make it private and add required friends
82 */
83 pipe_reservation(detail::sycl_2_2::pipe_reservation<accessor_detail> &&pr)
84 : implementation {
85 new detail::sycl_2_2::pipe_reservation<accessor_detail> { std::move(pr) } }
86 {}
87
88
89 /** Test if the pipe_reservation has been correctly allocated
90
91 \return true if the pipe_reservation can be used and committed
92 */
93 operator bool() const {
94 return *implementation;
95 }
96
97
98 /// Get the number of reserved element(s)
99 std::size_t size() const {
100 return implementation->size();
101 }
102
103
104 /// Access to a given element of the reservation
105 reference operator[](std::size_t index) const {
106 return (*implementation)[index];
107 }
108
109
110 /** Force a commit operation
111
112 Normally the commit is implicitly done in the destructor, but
113 sometime it is useful to do it earlier.
114 */
115 void commit() const {
116 return implementation->commit();
117 }
118
119
120 /// Get an iterator on the first element of the reservation station
121 iterator begin() const {
122 return implementation->begin();
123 }
124
125
126 /// Get an iterator past the end of the reservation station
127 iterator end() const {
128 return implementation->end();
129 }
130
131
132 /// Build a constant iterator on the first element of the reservation station
133 const_iterator cbegin() const {
134 return implementation->begin();
135 }
136
137
138 /// Build a constant iterator past the end of the reservation station
139 const_iterator cend() const {
140 return implementation->end();
141 }
142
143
144 /// Get a reverse iterator on the last element of the reservation station
145 reverse_iterator rbegin() const {
146 return std::make_reverse_iterator(end());
147 }
148
149
150 /** Get a reverse iterator on the first element past the end of the
151 reservation station */
152 reverse_iterator rend() const {
153 return std::make_reverse_iterator(begin());
154 }
155
156
157 /** Get a constant reverse iterator on the last element of the
158 reservation station */
159 const_reverse_iterator crbegin() const {
160 return std::make_reverse_iterator(cend());
161 }
162
163
164 /** Get a constant reverse iterator on the first element past the
165 end of the reservation station */
166 const_reverse_iterator crend() const {
167 return std::make_reverse_iterator(cbegin());
168 }

170};
