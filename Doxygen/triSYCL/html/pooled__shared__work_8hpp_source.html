12// It provides a scheduler sharing some work across a given number of
13// threads.
14//
15// The fiber pool can be stopped and several instances can coexist in
16// the same program.
17//
18// The main change to the original scheduler is to replace static
19// variables by a shared context.

97 void awakened(bf::context * ctx) noexcept override {
98 if (ctx->is_context(bf::type::pinned_context)) { /*<
99 recognize when we're passed this thread's main fiber (or an
100 implicit library helper fiber): never put those on the shared
101 queue
102 >*/
103 lqueue_.push_back(*ctx);
104 } else {
105 ctx->detach();
106 std::unique_lock lk { pool_ctx_->rqueue_mtx_ }; /*<
107 worker fiber, enqueue on shared queue
108 >*/
109 pool_ctx_->rqueue_.push_back(ctx);
110 }
111 }
112
113
114 bf::context * pick_next() noexcept override {
115 bf::context * ctx = nullptr;
116 std::unique_lock lk { pool_ctx_->rqueue_mtx_ };
117 auto &rq = pool_ctx_->rqueue_;
118 if (!rq.empty()) { /*<
119 pop an item from the ready queue
120 >*/
121 ctx = rq.front();
122 rq.pop_front();
123 lk.unlock();
124 BOOST_ASSERT(nullptr != ctx);
125 bf::context::active()->attach(ctx); /*<
126 attach context to current scheduler via the active fiber
127 of this thread
128 >*/
129 } else {
130 lk.unlock();
131 if (!lqueue_.empty()) { /*<
132 nothing in the ready queue, return main or dispatcher fiber
133 >*/
134 ctx = & lqueue_.front();
135 lqueue_.pop_front();
136 }
137 }
138 return ctx;
139 }
140
141
142 bool has_ready_fibers() const noexcept override {
143 std::unique_lock lock { pool_ctx_->rqueue_mtx_ };
144 return !pool_ctx_->rqueue_.empty() || !lqueue_.empty();
145 }
146
147
148 void suspend_until(std::chrono::steady_clock::time_point const& time_point)
149 noexcept override {
150 if (pool_ctx_->suspend_) {
151 if (std::chrono::steady_clock::time_point::max() == time_point) {
152 std::unique_lock lk { mtx_ };
153 cnd_.wait(lk, [this](){ return flag_; });
154 flag_ = false;
155 } else {
156 std::unique_lock lk { mtx_ };
157 cnd_.wait_until(lk, time_point, [this](){ return flag_; });
158 flag_ = false;
159 }
160 }
161 }
162
163
164 void notify() noexcept override {
165 if (pool_ctx_->suspend_) {
166 std::unique_lock lk { mtx_ };
167 flag_ = true;
168 lk.unlock();
169 cnd_.notify_all();
170 }
171 }

173};
