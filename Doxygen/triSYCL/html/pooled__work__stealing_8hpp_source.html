11// It provides a work stealing scheduler across a given number of threads.
12//
13// The fiber pool can be stopped and several instances can coexist in
14// the same program.
15//
16// The main change to the original scheduler is to replace static
17// variables by a shared context.

120 void awakened(boost::fibers::context * ctx) noexcept override {

127 bf::context * pick_next() noexcept override {
128 bf::context * victim = rqueue_.pop();
129 if (nullptr != victim) {
130 boost::context::detail::prefetch_range(victim, sizeof(*victim));
131 if (!victim->is_context(bf::type::pinned_context)) {
132 bf::context::active()->attach(victim);
133 }
134 }
135 else {
136 std::size_t size = pool_ctx_->thread_count_;
137 // Work stealing is only possible with more than 1 thread
138 if (BOOST_LIKELY(size > 1)) {
139 std::size_t count = 0;
140 static thread_local std::minstd_rand generator { std::random_device{}() };
141 std::uniform_int_distribution< std::uint32_t > distribution
142 { 0, static_cast<std::uint32_t>(size - 1) };
143 do {
144 std::uint32_t id = 0;
145 do {
146 ++count;
147 // Random selection of one logical CPU that belongs to the
148 // local NUMA node
149 id = distribution(generator);
150 // Prevent stealing from own scheduler
151 } while (id == id_);
152 // Steal context from other scheduler
153 victim = pool_ctx_->schedulers_[id]->steal();
154 } while (nullptr == victim && count < size);
155 if (nullptr != victim) {
156 boost::context::detail::prefetch_range(victim, sizeof(*victim));
157 BOOST_ASSERT(!victim->is_context(bf::type::pinned_context));
158 bf::context::active()->attach(victim);
159 }
160 }
161 }
162 return victim;
163 }

176 void suspend_until(std::chrono::steady_clock::time_point const& time_point)
177 noexcept override {
178 if (pool_ctx_->suspend_) {
179 if (std::chrono::steady_clock::time_point::max() == time_point) {
180 std::unique_lock lk { mtx_ };
181 cnd_.wait(lk, [&] { return flag_; });
182 flag_ = false;
183 }
184 else {
185 std::unique_lock lk { mtx_ };
186 cnd_.wait_until(lk, time_point, [&] { return flag_; });
187 flag_ = false;
188 }
189 }
190 }
191
192
193 void notify() noexcept override {
194 if (pool_ctx_->suspend_) {
195 std::unique_lock lk { mtx_ };
196 flag_ = true;
197 lk.unlock();
198 cnd_.notify_all();
199 }
200 }
