21/** Provide an implementation as shared_ptr with total ordering and
22 hashing to be used with algorithms and in (un)ordered containers
23
24 To be used, a Parent class wanting an Implementation needs to
25 inherit from.
26
27 The implementation ends up in a member really named
28 "implementation".
29
30 \code
31 public detail::shared_ptr_implementation<Parent, Implementation>
32 \endcode
33
34 and also inject in std namespace a specialization for
35 \code hash<Parent> \endcode
36*/
37template <typename Parent, typename Implementation>
38struct shared_ptr_implementation : public boost::totally_ordered<Parent> {
39
40 /// The implementation forward everything to this... implementation
41 std::shared_ptr<Implementation> implementation;
42
43 /// The implementation directly as a shared pointer
44 shared_ptr_implementation(std::shared_ptr<Implementation> i)
45 : implementation { i } {}
46
47
48 /// The implementation takes the ownership from a raw pointer
49 shared_ptr_implementation(Implementation *i) : implementation { i } {}
50
51
52 /// Keep all other constructors to have usual shared_ptr behaviour
53 shared_ptr_implementation() = default;
