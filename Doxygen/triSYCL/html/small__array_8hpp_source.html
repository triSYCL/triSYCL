118/** Define a multi-dimensional index, used for example to locate a work
119 item or a buffer element
120
121 Unfortunately, even if std::array is an aggregate class allowing
122 native list initialization, it is no longer an aggregate if we derive
123 from an aggregate. Thus we have to redeclare the constructors.
124
125 \param BasicType is the type element, such as int
126
127 \param Dims is the dimension number, typically between 1 and 3
128
129 \param FinalType is the final type, such as range<> or id<>, so that
130 boost::operator can return the right type
131
132 std::array<> provides the collection concept, with .size(), == and !=
133 too.
134
135 Can't use boost::shiftable, as it doesn't define the U, T overload
136 outside the class, but when that is added, it gets ambiguity problems.
137*/
138template <typename BasicType,
139 typename FinalType,
140 std::size_t Dims>
141struct small_array : std::array<BasicType, Dims>,
142 // To have all the usual arithmetic operations on this type and bitwise
143 // operations. Note these operation classes are chained via:
144 // https://www.boost.org/doc/libs/1_69_0/libs/utility/operators.htm#chaining
145 // without the chaining the empty base class optimization will break and the
146 // class will bloat other types like vec.hpp
147 boost::euclidean_ring_operators<FinalType,
148 boost::euclidean_ring_operators<FinalType, BasicType,
149 boost::bitwise<FinalType,
150 boost::bitwise<FinalType, BasicType>>>>,
151 // Already provided by array<> lexicographically:
152 // boost::equality_comparable<FinalType>,
153 // boost::less_than_comparable<FinalType>,
154 // Add a display() method
155 detail::display_vector<FinalType>,
156 // Just to find easily this is a small_array without templates
157 detail::small_array_tag {
158 /* Note that constexpr size() from the underlying std::array provides
159 the same functionality */
160 static constexpr size_t dimension = Dims;
161
162 using element_type = BasicType;
163
164 /** A constructor from another array
165
166 Make it explicit to avoid spurious range<> constructions from int *
167 for example
168 */
169 template <typename SourceType>
170 small_array(const SourceType src[Dims]) {
171 // (*this)[0] is the first element of the underlying array
172 std::copy_n(src, Dims, &(*this)[0]);
173 }
174

176 /// A constructor from another small_array of the same size
177 template <typename SourceBasicType,
178 typename SourceFinalType>
179 small_array(const small_array<SourceBasicType,
180 SourceFinalType,
181 Dims> &src) {
182 std::copy_n(&src[0], Dims, &(*this)[0]);
183 }

185 /** Initialize the array from a list of exactly Dims elements
186
187 Strangely, even when using the array constructors, the
188 initialization of the aggregate is not available. So recreate an
189 equivalent here.
190 */
191 template <typename... Types>
192 requires(sizeof...(Types) == Dims)
193 small_array(const Types&... args)
194 : std::array<BasicType, Dims> {
195 // Allow a loss of precision in initialization with the static_cast
196 { static_cast<BasicType>(args)... }
197 } {}

199 /// Construct a small_array from a std::array
200 template <typename SourceBasicType>
201 small_array(const std::array<SourceBasicType, Dims> &src)
202 : std::array<BasicType, Dims>(src) {}

205 /// Keep other constructors from the underlying std::array
206 using std::array<BasicType, Dims>::array;

208 /// Keep the synthesized constructors

211 /// Return the element of the array
212 auto get(std::size_t index) const {

321/** A small array to implement SYCL objects like id, range,
322 item... with 1, 2 or 3 elements (or even more if
323 detail::allow_any_dimension is true), with the implicit
324 constructors */
325template <typename BasicType, typename FinalType, std::size_t Dims>
326struct small_array_sycl : small_array<BasicType, FinalType, Dims> {
327 static_assert(1 <= Dims, "SYCL dimensions are greater than or equal to 1");
328 static_assert(detail::allow_any_dimension || Dims <= 3,
329 "SYCL dimensions are less than or equal to 3");
330
331 /// Keep other constructors
332 small_array_sycl() = default;
333
334 /** Construct from a range-like object inheriting from a
335 compatible std::array
336
337 This useful in implementation details for example to construct a
338 sycl::id from a sycl::range.
339 */
340 template <typename Range>
341 requires requires {
342 ranges::copy(std::declval<Range>(), std::declval<BasicType*>());
343 }
344 small_array_sycl(Range&& r) {
345 // Copy to internal value
346 ranges::copy(std::forward<Range>(r), this->begin());
347 }
348
349 /** Constructor taking either Dims arguments compatible with BasicType
350 or just 1 argument compatible with BasicType and broadcast its
351 value to all the elements */
352 template <typename... Args>
353 // Check that each parameter is convertible to the BasicType
354 // Full C++20: requires (std::convertible_to<Args, BasicType> && ...)
355 // In the meantime:
356 requires requires { (static_cast<BasicType>(std::declval<Args>()), ...); }
357 small_array_sycl(Args... args) {
358 if constexpr (sizeof...(Args) == Dims) {
359 // If there is the right number of elements we initialize each
360 // element from each parameter, allowing some narrowing
361 static_cast<std::array<BasicType, Dims>&>(
362 *this) = { static_cast<BasicType>(args)... };
363 } else if constexpr (sizeof...(Args) == 1) {
364 // If there is only 1 argument, this is a broadcast
365 detail::assign_from_tuple(
366 static_cast<std::array<BasicType, Dims>&>(*this),
367 detail::expand_to_vector<small_array_sycl, std::tuple<BasicType>,
368 true> {}
369 .expand(args...));
370 } else
371 static_assert(true, "There should be either 1 argument to broadcast the "
372 "value or 1 value per dimension");
373 }
374
375 /** Conversion operator to scalar BasicType when only 1 dimension
376
377 This is used for example to implicitly convert a sycl::id<1> to
378 a std::size_t
379 */
380 template <auto T = Dims>
381 requires(Dims == 1)
382 operator BasicType() {
383 return this->get(0);
384 }
385};
