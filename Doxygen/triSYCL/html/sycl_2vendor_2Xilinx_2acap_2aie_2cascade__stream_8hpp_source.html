46struct cascade_stream {
47 using geo = Geography;
48 /** The pipes for the cascade streams, with 1 spare pipe on each
49 side of the PE string
50
51 \todo Use a data type with 384 bits
52 */
53 ::trisycl::sycl_2_2::static_pipe<int, 4>
54 cascade_stream_pipes[geo::x_size*geo::y_size + 1];
55
56 /* Cascade stream layout
57
58 On even rows, a tile use cascade_stream_pipes[y][x] as input and
59 cascade_stream_pipes[y][x + 1] as output
60
61 On odd rows the flow goes into the other direction, so a tile
62 use cascade_stream_pipes[y][x + 1] as input and
63 cascade_stream_pipes[y][x] as output
64 */
65
66 /** Get a blocking read accessor to the cascade stream input
67
68 \param T is the data type used to read from the cascade
69 stream pipe
70
71 \param[in] x is the horizontal tile coordinate
72
73 \param[in] y is the vertical tile coordinate
74 */
75 template <typename T>
76 auto get_cascade_stream_in(int x, int y) const {
77 return cascade_stream_pipes[geo::cascade_linear_id(x, y)]
78 .template get_access<access::mode::read,
79 access::target::blocking_pipe>();
80 }
81
82
83 /** Get a blocking write accessor to the cascade stream output
84
85 \param T is the data type used to write to the cascade
86 stream pipe
87
88 \param[in] x is the horizontal tile coordinate
89
90 \param[in] y is the vertical tile coordinate
91 */
92 template <typename T>
93 auto get_cascade_stream_out(int x, int y) const {
94 // The output is connected to the down-stream neighbour of the cascade
95 return cascade_stream_pipes[geo::cascade_linear_id(x, y) + 1]
96 .template get_access<access::mode::write,
97 access::target::blocking_pipe>();
98 }
99
100};
