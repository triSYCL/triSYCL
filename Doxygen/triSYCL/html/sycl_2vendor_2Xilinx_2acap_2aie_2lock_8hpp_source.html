37struct lock_unit {
38 // There are 16 hardware locks per memory tile
39 auto static constexpr lock_number = 16;
40
41 using lock_id_t = decltype(lock_number);
42
43 /// The individual locking system
44 struct locking_device {
45 /// The type of the value stored in the locking device
46 using value_t = bool;
47
48 /* The problem here is that \c mutex and \c condition_variable
49 are not moveable while the instantiation of a memory module uses
50 move assignment with Boost.Hana...
51
52 So allocate them dynamically and keep them in a \c std::unique_ptr
53 so globally the type is moveable */
54
55 /// The mutex to provide the basic protection mechanism
56 std::unique_ptr<boost::fibers::mutex> m { new boost::fibers::mutex { } };
57
58 /// The condition variable to wait/notify for some value
59 std::unique_ptr<boost::fibers::condition_variable> cv {
60 new boost::fibers::condition_variable { } };
61
62 /// The value to be waited for, initialized to false on reset
63 value_t value = false;
64
65 /// Lock the mutex
66 void acquire() {
67 m->lock();
68 }
69
70
71 /// Unlock the mutex
72 void release() {
73 m->unlock();
74 }
75
76
77 /// Wait until the internal value has the expectation
78 void acquire_with_value(value_t expectation) {
79 std::unique_lock lk { *m };
80 cv->wait(lk, [&] { return expectation == value; });
81 }
82
83
84 /// Release and update with a new internal value
85 void release_with_value(value_t new_value) {
86 {
87 std::unique_lock lk { *m };
88 value = new_value;
89 }
90 // By construction there should be only one client waiting for it
91 cv->notify_one();
92 }
93 };
94
95 /// The locking units of the locking device
96 locking_device locks[lock_number];
97
98 /// Get the requested lock
99 auto &lock(int i) {
100 assert(0 <= i && i < lock_number);
101 return locks[i];
102 }
103
104};
