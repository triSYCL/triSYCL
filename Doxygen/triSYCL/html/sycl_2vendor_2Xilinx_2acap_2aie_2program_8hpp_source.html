176 /// Create the AIE program with the tiles and memory modules
177 program(AIEDevice &aie_d) : aie_d { aie_d } {
178 boost::hana::for_each(tiles, [&] (auto& t) {
179 // Inform each tile about its program
180 t.set_program(*this);
181 // Inform each tile about their tile infrastructure
182 t.set_tile_infrastructure(aie_d.tile(t.x, t.y));
183 // Keep track of each base tile
184 tile_bases[t.y][t.x] = &t;
185 });
186 // Connect each memory to its infrastructure
187 boost::hana::for_each(memory_modules, [&] (auto& m) {
188 // Inform each tile about their tile infrastructure
189 m.set_memory_infrastructure(aie_d.mem(m.x, m.y));
190 // Keep track of each base tile
191 memory_modules_bases[m.y][m.x] = &m;
192 });
193 }

211 /// Wait for the end of the execution of each tile
212 void wait() {
213 boost::hana::for_each(tiles, [&] (auto& t) {
214 TRISYCL_DUMP_T("Joining AIE tile (" << t.x << ',' << t.y << ")...");
215 t.wait();

220 /** Launch the programs of all the tiles of the CGRA in their own
221 executor (CPU thread, fiber...) and wait for their completion.
222
223 This is the main member function to use to launch the execution.
224 */
225 void run() {
226 // Start each tile program in its own executor
227 boost::hana::for_each(tiles, [&] (auto& t) {
228 t.single_task([&] {
229 TRISYCL_DUMP_T("Starting AIE tile (" << t.x << ',' << t.y
230 << ") linear id = " << t.linear_id());
231 /* Just use a capture by reference in the following
232 because there is direct execution here */
233 auto kernel = [&] {
234 // If the tile has an operator(), use it
235 if constexpr (requires { t(); })
236 return [&] { t(); };
237 /* Else the kernel should have a run
238 member function and use it. */
239 else
240 return [&] { t.run(); };
241 }();
242 using kernel_type = decltype(kernel);
243 // Use the kernel type as its SYCL name too
244 kernel_outliner<kernel_type, kernel_type>(kernel);
245 TRISYCL_DUMP_T("Stopping AIE tile (" << t.x << ',' << t.y << ')');
246 });
247 });
248 wait();
249 }

251 /** Run synchronously an heterogeneous invocable collectively on the device
252
253 \param f is an invocable taking an heterogeneous tile handler
254
255 \todo Factorize out the 2 run functions
256 */
257 template <typename Invocable>
258 void run(Invocable&& f) {
259 // Start each tile program in its own executor
260 boost::hana::for_each(tiles, [&] (auto& t) {
261 t.single_task([&] {
262 TRISYCL_DUMP_T("Starting AIE tile (" << t.x << ',' << t.y
263 << ") linear id = " << t.linear_id());
264 /// Each tile gets its own copy of work
265 auto kernel = [&, work = f] { work(t); };
266 using kernel_type = decltype(kernel);
267 // Use the kernel type as its SYCL name too
268 kernel_outliner<kernel_type, kernel_type>(kernel);
269 TRISYCL_DUMP_T("Stopping AIE tile (" << t.x << ',' << t.y << ')');
270 });
271 });
272 wait();
273 }
