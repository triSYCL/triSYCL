165 /// Launch a callable on this tile
166 template <typename Work> auto& single_task(Work&& f) {
167 /* Add an immediate lambda call to avoid a warning about capturing
168 this when non using it */
169 auto kernel = [&] {
170 if constexpr (requires { f(); })
171 /* If the invocable is not interested by the handler, do not
172 provide it. Use a copy of the callable to make the
173 separation from host to device memory very clear. */
174 return [work = f] () mutable { return work(); };
175 else
176 /** Recycle the tile_infrastructure shared_ptr as a fake tile
177 handle which is copyable.
178
179 \todo In a device implementation we should have a real
180 tile_handler type making sense on the device instead of
181 just *this */
182 return [this, work = f] () mutable { return work(*this); };
183 }();
184
185 implementation->single_task(kernel);
186 // To allow chaining commands
187 return *this;
188 }
189
190 /// Wait for the execution of the callable on this tile
191 auto& wait() {
192 implementation->wait();
193 // To allow chaining commands
194 return *this;
195 }
