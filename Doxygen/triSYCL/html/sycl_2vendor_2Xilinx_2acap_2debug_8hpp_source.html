35/** Verify that an SPMD program has a bulk-synchronous behaviour
36 across all the tiles of the CGRA.
37
38 If there are some discrepancies, display the tile id which is too
39 late or too early.
40*/
41template <typename Geography> class bsp_checker {
42 /// A vector clock to compare differerent tiles together
43 std::array<std::atomic<int>, Geography::size> vector_clock_alloc;
44 std::mdspan<std::atomic<int>,
45 std::extents<std::size_t, Geography::x_size, Geography::y_size>>
46 vector_clock { &vector_clock_alloc[0] };
47
48 public:
49 /** Check that none of the tiles executing this are separated by
50 more than one execution step each other.
51
52 This functions has to be called in a BSP synchronous way by all
53 the tiles.
54 */
55 void check(int x, int y) {
56 ++vector_clock[x, y];
57 /* While the algorithm execution is not atomic, it is enough to
58 find some asynchronous bugs */
59 auto [min_element, max_element] = std::minmax_element(
60 vector_clock_alloc.cbegin(), vector_clock_alloc.cend());
61
62 if (*max_element - *min_element > 1) {
63 TRISYCL_DUMP_T(std::dec << "compute(" << x << ',' << y
64 << ") vector clock min = " << *min_element
65 << ", max = " << *max_element);
66 std::terminate();
67 }
68 }
69};
