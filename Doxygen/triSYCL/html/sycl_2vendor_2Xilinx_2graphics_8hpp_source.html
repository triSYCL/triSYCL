141/** Color palette to project a linear data space to an RGB color space
142
143 Reimplement the palette concept by Nicolas Paris in PompC & HyperC
144 back from the 90's... :-)
145*/
146struct palette {
147 static constexpr int size = 256;
148 std::array<rgb, size> color_mapping;
149
150 enum kind { gray, rainbow };
151 kind k;
152 int phase;
153 int clip;
154 int frequency_log2;
155
156 /** Create a palette
157
158 \param[in] k is the kind of palette
159
160 \param[in] phase is optional and allows the palette to have a
161 phase shift instead of starting at 0
162
163 \param[in] clip is optional and specifies a value to be enhanced
164 */
165 palette(kind k = gray, int phase = 0, int frequency_log2 = 0, int clip = -1)
166 : k { k }, phase { phase } , clip { clip }, frequency_log2{frequency_log2} {
167 update();
168 }
169
170
171 // Create a gray palette
172 void gray_palette() {
173 for (int i = 0; auto &e : color_mapping) {
174 if (i == clip)
175 // Put a full red color for the selected value
176 e = { 255, 0, 0 };
177 else {
178 int j;
179 // There is no bidirectional shift operator in C++
180 if (frequency_log2 < 0)
181 // Divide the step by 2^{-fl}
182 j = i >> -frequency_log2;
183 else
184 // Multiply the step by 2^{fl}
185 j = i << frequency_log2;
186 auto v = static_cast<std::uint8_t>(j + phase);
187 // A non-saturated "gray" of luminosity v
188 e = { v, v, v };
189 }
190 ++i;
191 }
192 }
193
194
195 // Create a rainbow palette
196 void rainbow_palette() {
197 Gdk::RGBA color;
198 for (int i = 0; auto &e : color_mapping) {
199 // Put the selected value as black
200 double luminosity = !(i == clip);
201 int j;
202 // There is no bidirectional shift operator in C++
203 if (frequency_log2 < 0)
204 // Divide the step by 2^{-fl}
205 j = i >> -frequency_log2;
206 else
207 // Multiply the step by 2^{fl}
208 j = i << frequency_log2;
209 auto v = static_cast<std::uint8_t>(j + phase);
210 color.set_hsv(360.*v/size, 1, luminosity);
211 e = { static_cast<std::uint8_t>(color.get_red_u() >> 8),
212 static_cast<std::uint8_t>(color.get_green_u() >> 8),
213 static_cast<std::uint8_t>(color.get_blue_u() >> 8) };
214 ++i;
215 }
216 }
217
218
219 /* Update the palette and enforce some value invariants for
220 defensive programming and GUI features */
221 void update() {
222 // Implement modulo arithmetic on interval [-1, size - 1]
223 if (clip < -1)
224 clip = size - 1;
225 else if (clip >= size)
226 clip = -1;
227 // The frequency can only evolve in this interval
228 frequency_log2 = std::clamp(frequency_log2, -8, 7);
229 /* The phase can evolve modulo size. When there are some negative
230 numbers it is implementation dependent but that is enough for
231 this GUI */
232 phase = phase % size;
233
234 switch (k) {
235 case gray:
236 gray_palette();
237 break;
238 case rainbow:
239 rainbow_palette();
240 break;
241 }
242 }

247 \param[in] new_k is the kind of palette

249 \param[in] new_phase allows the palette to have a phase shift

254 void set(kind new_k, int new_phase, int new_frequency_log2, int new_clip) {

264 */
265 rgb palettize(double data, double min_value, double max_value) {
266 std::uint8_t v = (data - min_value)*255/(max_value - min_value);
267 // Write the same value for RGB to have a grey level
268 return color_mapping[v];

308 /// Handle with any key pressed
309 bool handle_key_press_event(GdkEventKey &event) {
310 /* Only listen to keys without modifiers (no control, alt...) but
311 still allows shift */
312 if (!(event.state
313 & gtk_accelerator_get_default_mod_mask()
314 & ~GDK_SHIFT_MASK)) {
315 switch (event.keyval) {
316 case GDK_KEY_0:
317 case GDK_KEY_KP_0:
318 k = gray;
319 update();
320 break;
321 case GDK_KEY_1:
322 case GDK_KEY_KP_1:
323 k = rainbow;
324 update();
325 break;
326 case GDK_KEY_minus:
327 case GDK_KEY_KP_Subtract:
328 case GDK_KEY_underscore:
329 decrease_frequency();
330 break;
331 case GDK_KEY_plus:
332 case GDK_KEY_KP_Add:
333 case GDK_KEY_equal:
334 case GDK_KEY_KP_Equal:
335 increase_frequency();
336 break;
337 case GDK_KEY_h:
338 increase_phase();
339 break;
340 case GDK_KEY_H:
341 increase_phase();
342 decrease_clip();
343 break;
344 case GDK_KEY_j:
345 decrease_clip();
346 break;
347 case GDK_KEY_k:
348 increase_clip();
349 break;
350 case GDK_KEY_l:
351 decrease_phase();
352 break;
353 case GDK_KEY_L:
354 decrease_phase();
355 increase_clip();
356 break;
357 default:
358 // Not a key press for us: let the other listeners to handle
359 return false;
360 }
361 // We have handled the event: do not go on with any other listener
362 return true;
363 }
364 // Not the right modifiers: let the other listeners to handle
365 return false;
366 }

371/// An application window displaying a grid of tiled images
372struct image_grid : frame_grid {
373 /// The linearized 2D vector of images
374 std::vector<Gtk::Image> images;
375
376 /// Horizontal image size in pixels
377 int image_x;
378
379 /// Vertical image size in pixels
380 int image_y;
381
382 /// The image pixel zooming factor for both dimension
383 int zoom;
384
385 /// Dispatcher to invoke something in the graphics thread in a safe way
386 Glib::Dispatcher dispatcher;
387
388 /// Protection against concurrent update
389 std::mutex dispatch_protection;
390
391 /// The condition variable used to queue up graphics update submission
392 std::condition_variable cv;
393
394 /// What to dispatch
395 std::function<void(void)> work_to_dispatch;
396
397 /// The RGB palette used to render the image values
398 palette p;
399
400 /// A global done state across all the tile to allow a global shutdown
401 bool done_snapshot;
402
403 /// Set to true once "done" state has been sampled
404 std::atomic_flag done_has_been_sampled;
405
406 /// Barrier to compute the global done state across the tiles
407 boost::barrier done_barrier { static_cast<unsigned int>(nx*ny) };
408
409 /** Create a grid of tiled images
410
411 \param[in] nx is the numer of tiles horizontally
412
413 \param[in] ny is the numer of tiles vertically
414
415 \param[in] image_x is the horizontal image pixel size
416
417 \param[in] image_y is the vertical image pixel size
418
419 \param[in] zoom is the zooming factor applied to image pixels,
420 both horizontally and vertically
421 */
422 image_grid(int nx, int ny, int image_x, int image_y, int zoom)
423 : frame_grid { nx , ny }
424 , image_x { image_x }
425 , image_y { image_y }
426 , zoom { zoom }
427 {
428 for (int y = 0; y < ny; ++y)
429 for (int x = 0; x < nx; ++x) {
430 auto &f = get_frame(x, y);
431 auto pb = Gdk::Pixbuf::create(Gdk::Colorspace::COLORSPACE_RGB
432 , false //< has_alpha
433 , 8 //< bits_per_sample
434 , image_x*zoom //< width
435 , image_y*zoom //< height
436 );
437 images.emplace_back(pb);
438 // Display the frame with the lower y to the South
439 f.add(images.back());
440 }
441
442 // Hook a generic dispatcher
443 dispatcher.connect([&] {
444 {
445 // Only 1 customer at a time
446 std::lock_guard lock { dispatch_protection };
447 // Skip the work when done to avoid dead lock
448 if (done)
449 // Wake-up everybody waiting for sending some work to
450 // realize that they have to give up on their hope
451 cv.notify_all();
452 else
453 work_to_dispatch();
454 // Discard the previous work
455 work_to_dispatch = nullptr;
456 }
457 // We can serve the next customer
458 cv.notify_one();
459 });
460
461 // Go live
462 show_all_children();
463 }
464
465
466 /// Deal with the key pressed
467 bool on_key_press_event(GdkEventKey* event) override {
468 // Forward to the palette for action
469 return p.handle_key_press_event(*event);
470 }
471
472 /// Submit some work to the graphics thread
473 void submit(std::function<void(void)> f) {
474 std::unique_lock lock { dispatch_protection };
475 // Wait for no work being dispatched or the end
476 cv.wait(lock, [&] { return !work_to_dispatch || done; } );
477 // Do not submit anything if we are in the shutdown process already
478 if (!done) {
479 work_to_dispatch = std::move(f);
480 // Ask the graphics thread for some work
481 dispatcher.emit();
482 }
483 };

486 /** Update the image of a tile
487
488 \param[in] x is the tile horizontal id
489
490 \param[in] y is the tile vertical id
491
492 \param[in] data is a 2D MDspan of extent at most image_y by
493 image_x. Only the pixels of the extents are drawn
494
495 \param[in] min_value is the value represented with minimum of
496 graphics palette color
497
498 \param[in] max_value is the value represented with maximum of
499 graphics palette color
500 */
501 template <typename MDspan, typename RangeValue,
502 /* Implement a cheap mdspan concept requirement, so that
503 this function is not called when it is a pointer*/
504 typename = typename MDspan::extents_type
505 >
506 void update_tile_data_image(int x, int y,
507 const MDspan &data,
508 RangeValue min_value,
509 RangeValue max_value) {
510 /* Painful: first I cannot use a std::unique_ptr because the
511 std::function below needs to be copy-constructible, then the
512 std::make_shared taking an array size comes only in C++20 while
513 it is available for std::make_unique in C++17... */
514 std::shared_ptr<std::uint8_t[]> d { new std::uint8_t[3*image_x*image_y] };
515 std::mdspan output { reinterpret_cast<rgb*>(d.get()), image_y, image_x };
516 // For each pixel of the md_span or of the image, which one is smaller
517 for (int j = 0; j < std::min(static_cast<int>(data.extent(0)), image_y);
518 ++j)
519 for (int i = 0; i < std::min(static_cast<int>(data.extent(1)), image_x);
520 ++i) {
521 /* Mirror the image vertically to display the pixels in a
522 mathematical sense */
523 output[image_y - 1 - j, i] =
524 p.palettize(data[j, i], min_value, max_value);
525 }
526 // Send the graphics updating code
527 submit([=, this] {
528 // Create a first buffer, allowing later zooming
529 auto pb = Gdk::Pixbuf::create_from_data(d.get()
530 , Gdk::Colorspace::COLORSPACE_RGB
531 , false//< has_alpha
532 , 8 //< bits_per_sample
533 , image_x //< width
534 , image_y //< height
535 , image_x*3 //< rowstride
536 );
537 // Update the pixel buffer of the image with some zooming
538 images.at(x + nx*y).set(pb->scale_simple(image_x*zoom,
539 image_y*zoom,
540 Gdk::INTERP_NEAREST));
541 });
542 }
543
544
545 /** Update the image of a tile of size image_y by image_x from a pointer
546
547 \param[in] x is the tile horizontal id
548
549 \param[in] y is the tile vertical id
550
551 \param[in] data is a pointer to the pixels to be drawns
552
553 \param[in] min_value is the value represented with minimum of
554 graphics palette color
555
556 \param[in] max_value is the value represented with maximum of
557 graphics palette color
558 */
559 template <typename DataType, typename RangeValue>
560 void update_tile_data_image(int x, int y,
561 const DataType *data,
562 RangeValue min_value,
563 RangeValue max_value) {
564 // Wrap the pointed area into an MDspan
565 const std::mdspan md { data, image_y, image_x };
566 update_tile_data_image(x, y, md, min_value, max_value);
567 }

576 /// Test if the window has been closed after synchronizing with a
577 /// barrier accross all the tiles
578 bool is_done_barrier() {
579 // Only 1 thread sample the graphics status
580 if (done_has_been_sampled.test_and_set())
581 done_snapshot = done;
582 // Then wait for everybody to synchronize
583 done_barrier.count_down_and_wait();
584 // Reinitialize the flag for the next iteration
585 done_has_been_sampled.clear();
586 return done_snapshot;
587 }

592/** A graphics application running in a separate thread to display
593 images in a grid of tiles */
594struct application {
595 std::thread t;
596 std::unique_ptr<graphics::image_grid> w;
597 bool initialized = false;
598
599 /** Start the graphics application
600
601 \param[inout] argc is the standard C program argument number
602
603 \param[inout] argv is the standard C program argument array
604
605 \param[in] nx is the numer of tiles horizontally
606
607 \param[in] ny is the numer of tiles vertically
608
609 \param[in] image_x is the horizontal image pixel size
610
611 \param[in] image_y is the vertical image pixel size
612
613 \param[in] zoom is the zooming factor applied to image pixels,
614 both horizontally and vertically
615
616 \return a reference on *this to allow operation chaining
617 */
618 auto& start(int &argc, char **&argv,
619 int nx, int ny, int image_x, int image_y, int zoom) {
620 // To be sure not passing over the asynchronous graphics start
621 std::promise<void> graphics_initialization;
622
623 /* Put all the graphics in its own thread. Since
624 Gtk::Application::create might modify argc and argv, capture by
625 reference. Since we have to wait for this thread, there should
626 not be a read from freed memory issue. */
627 t = std::thread { [&]() mutable {
628 // An application allowing several instance running at the same time
629 auto a =
630 Gtk::Application::create(argc, argv, "com.xilinx.trisycl.graphics",
631 Gio::APPLICATION_NON_UNIQUE);
632 /* Create the graphics object in this thread so the dispatcher
633 is bound to this thread too */
634 w.reset(new graphics::image_grid { nx, ny, image_x, image_y, zoom });
635 w->set_close_action([&] {
636 w->done = true;
637 });
638 // OK, the graphics system is in a usable state, unleash the main thread
639 graphics_initialization.set_value();
640 // Launch the graphics event loop handling with the graphics life
641 a->run(*w);
642
643 // Advertise that the graphics is shutting down
644 w->done = true;
645 } };
646 // Wait for the graphics to start
647 graphics_initialization.get_future().get();
648
649 return *this;
650 }

655 t.join();
656 }
