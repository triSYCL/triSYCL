33template <int Nbits = 32>
34struct xorshift {
35 /// Bit size of the random generator data type
36 static auto constexpr bit_size = Nbits;
37
38 /// The default initial internal state of the pseudo random generator
39 static auto constexpr initial_state = [] {
40 if constexpr (bit_size == 32) {
41 /* Default value for xor32 from Marsaglia,
42 p. 4 Section "3 Application to Xorshift RNGs" */
43 return std::uint32_t { 2463534242 };
44 } else if constexpr (bit_size == 64) {
45 /* Default value for xor32 from Marsaglia,
46 p. 4 Section "3 Application to Xorshift RNGs" */
47 return std::uint64_t { 88172645463325252 };
48 } else if constexpr (bit_size == 128) {
49 /// Default value for xor128 from Marsaglia, p. 5 Section "4 Summary"
50 return std::array<std::uint32_t, 4>
51 { 123456789, 362436069, 521288629, 88675123 };
52 } else
53 return nullptr;
54 }();
55
56 /// The type of the internal state of the generator
57 using value_type = std::remove_const_t<decltype(initial_state)>;
58
59 /// The type of the result
60 using result_type = value_type;
61
62 static_assert(!std::is_same_v<value_type, std::nullptr_t>,
63 "Bit size not implemented");
64
65 /// The internal state of the pseudo random generator
66 value_type state = initial_state;
67
68
69 /// The minimum returned value
70 static auto constexpr min() {
71 // It cannot return 0
72 return std::numeric_limits<result_type>::min() + 1;
73 };
74
75
76 /// The maximum returned value
77 static auto constexpr max() {
78 return std::numeric_limits<result_type>::max();
79 };
80
81
82 /** Initialize the internal state from a user-given value
83
84 Do not use the value 0 or the output is always 0
85 */
86 xorshift(const value_type& s)
87 : state { s }
88 {}
89
90
91 xorshift() = default;
92
93
94 /// Compute a new pseudo random integer
95 const result_type& operator()() {
96 if constexpr (bit_size == 32) {
97 /* Pick the one of type "I" with best bit equidistribution from
98 Panneton & L'Écuyer, Section "5.1 Equidistribution
99 properties"
100
101 X4 = (I + Ra )(I + Lb )(I + Rc )
102 */
103 state ^= state >> 7;
104 state ^= state << 1;
105 state ^= state >> 9;
106 }
107 else if constexpr (bit_size == 64) {
108 /* The xor64 from Marsaglia, p. 4 Section "3 Application to
109 Xorshift RNGs" */
110 state ^= state << 13;
111 state ^= state >> 7;
112 state ^= state << 17;
113 }
114 else if constexpr (bit_size == 128) {
115 // The xor128 from Marsaglia, p. 5 Section "4 Summary"
116 auto t = state[0]^(state[0]<<11);
117 state[0] = state[1];
118 state[1] = state[2];
119 state[2] = state[3];
120 state[3] = state[3]^(state[3]>>19)^(t^(t>>8));
121 }
122 else
123 // Just to error in that case
124 return nullptr;
125
126 return state;
127 }
128
129};
