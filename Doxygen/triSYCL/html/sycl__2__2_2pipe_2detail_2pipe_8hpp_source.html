139 /** Get the current number of elements in the pipe that can be read

148 TRISYCL_DUMP_T("size() cb.size() = " << cb.size()
149 << " cb.end() = " << (void *)&*cb.end()
150 << " reserved_for_reading() = " << reserved_for_reading()
151 << " reserved_for_writing() = " << reserved_for_writing());
152 /* The actual number of available elements depends from the

211 /** Try to write a value to the pipe
212
213 \param[in] value is what we want to write
214
215 \param[in] blocking specify if the call wait for the operation
216 to succeed
217
218 \return true on success
219
220 \todo provide a && version
221 */
222 bool write(const T &value, bool blocking = false) {
223 // Lock the pipe to avoid being disturbed
224 std::unique_lock<std::mutex> ul { cb_mutex };
225 TRISYCL_DUMP_T("Write pipe full = " << full()
226 << " value = " << value);
227
228 if (blocking)
229 /* If in blocking mode, wait for the not full condition, that
230 may be changed when a read is done */
231 read_done.wait(ul, [&] { return !full(); });
232 else if (full())
233 return false;
234
235 cb.push_back(value);
236 TRISYCL_DUMP_T("Write pipe front = " << cb.front()
237 << " back = " << cb.back()
238 << " cb.begin() = " << (void *)&*cb.begin()
239 << " cb.size() = " << cb.size()
240 << " cb.end() = " << (void *)&*cb.end()
241 << " reserved_for_reading() = " << reserved_for_reading()
242 << " reserved_for_writing() = " << reserved_for_writing());
243 // Micro-optimization: unlock before the notification
244 // https://en.cppreference.com/w/cpp/thread/condition_variable/notify_all
245 ul.unlock();
246 // Notify the clients waiting to read something from the pipe
247 write_done.notify_all();
248 return true;
249 }

252 /** Try to read a value from the pipe
253
254 \param[out] value is the reference to where to store what is
255 read
256
257 \param[in] blocking specify if the call wait for the operation
258 to succeed
259
260 \return true on success
261 */
262 bool read(T &value, bool blocking = false) {
263 // Lock the pipe to avoid being disturbed
264 std::unique_lock<std::mutex> ul { cb_mutex };
265 TRISYCL_DUMP_T("Read pipe empty = " << empty());
266
267 if (blocking)
268 /* If in blocking mode, wait for the not empty condition, that
269 may be changed when a write is done */
270 write_done.wait(ul, [&] { return !empty(); });
271 else if (empty())
272 return false;
273
274 TRISYCL_DUMP_T("Read pipe front = " << cb.front()
275 << " back = " << cb.back()
276 << " reserved_for_reading() = " << reserved_for_reading());
277 if (read_reserved_frozen)
278 /** If there is a pending reservation, read the next element to
279 be read and update the number of reserved elements */
280 value = cb.begin()[read_reserved_frozen++];
281 else {
282 /* There is no pending read reservation, so pop the read value
283 from the pipe */
284 value = cb.front();
285 cb.pop_front();
286 }
287
288 TRISYCL_DUMP_T("Read pipe value = " << value);
289 // Micro-optimization: unlock before the notification
290 // https://en.cppreference.com/w/cpp/thread/condition_variable/notify_all
291 ul.unlock();
292 // Notify the clients waiting for some room to write in the pipe
293 read_done.notify_all();
294 return true;
295 }

298 /** Compute the amount of elements blocked by read reservations, not yet
299 committed

301 This includes some normal reads to pipes between/after
302 un-committed reservations

311 /** Compute the amount of elements blocked by write reservations, not yet
312 committed
313
314 This includes some normal writes to pipes between/after
315 un-committed reservations
316
317 This function assumes that the data structure is locked
318 */
319 std::size_t reserved_for_writing() const {
320 if (w_rid_q.empty())
321 // No on-going reservation
322 return 0;
323 else
324 /* The reserved size is from the first element of the first
325 on-going reservation up to the end of the pipe content */
326 return cb.end() - w_rid_q.front().start;
327 }

330 /** Reserve some part of the pipe for reading
331
332 \param[in] s is the number of element to reserve
333
334 \param[out] rid is an iterator to a description of the
335 reservation that has been done if successful
336
337 \param[in] blocking specify if the call wait for the operation
338 to succeed
339
340 \return true if the reservation was successful
341 */
342 bool reserve_read(std::size_t s,
343 rid_iterator &rid,
344 bool blocking = false) {
345 // Lock the pipe to avoid being disturbed
346 std::unique_lock<std::mutex> ul { cb_mutex };
347
348 TRISYCL_DUMP_T("Before read reservation cb.size() = " << cb.size()
349 << " size() = " << size());
350 if (s == 0)
351 // Empty reservation requested, so nothing to do
352 return false;
353
354 if (blocking)
355 /* If in blocking mode, wait for enough elements to read in the
356 pipe for the reservation. This condition can change when a
357 write is done */
358 write_done.wait(ul, [&] { return s <= size(); });
359 else if (s > size())
360 // Not enough elements to read in the pipe for the reservation
361 return false;
362
363 // Compute the location of the first element of the reservation
364 auto first = cb.begin() + read_reserved_frozen;
365 // Increment the number of frozen elements
366 read_reserved_frozen += s;
367 /* Add a description of the reservation at the end of the
368 reservation queue */
369 r_rid_q.emplace_back(first, s);
370 // Return the iterator to the last reservation descriptor
371 rid = r_rid_q.end() - 1;
372 TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
373 << " size() = " << size());
374 return true;
375 }

378 /** Reserve some part of the pipe for writing
379
380 \param[in] s is the number of element to reserve
381
382 \param[out] rid is an iterator to a description of the
383 reservation that has been done if successful
384
385 \param[in] blocking specify if the call wait for the operation
386 to succeed
387
388 \return true if the reservation was successful
389 */
390 bool reserve_write(std::size_t s,
391 rid_iterator &rid,
392 bool blocking = false) {
393 // Lock the pipe to avoid being disturbed
394 std::unique_lock<std::mutex> ul { cb_mutex };
395
396 TRISYCL_DUMP_T("Before write reservation cb.size() = " << cb.size()
397 << " size() = " << size());
398 if (s == 0)
399 // Empty reservation requested, so nothing to do
400 return false;
401
402 if (blocking)
403 /* If in blocking mode, wait for enough room in the pipe, that
404 may be changed when a read is done. Do not use a difference
405 here because it is only about unsigned values */
406 read_done.wait(ul, [&] { return cb.size() + s <= capacity(); });
407 else if (cb.size() + s > capacity())
408 // Not enough room in the pipe for the reservation
409 return false;
410
411 /* If there is enough room in the pipe, just create default values
412 in it to do the reservation */
413 for (std::size_t i = 0; i != s; ++i)
414 cb.push_back();
415 /* Compute the location of the first element a posteriori since it
416 may not exist a priori if cb was empty before */
417 auto first = cb.end() - s;
418 /* Add a description of the reservation at the end of the
419 reservation queue */
420 w_rid_q.emplace_back(first, s);
421 // Return the iterator to the last reservation descriptor
422 rid = w_rid_q.end() - 1;
423 TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
424 << " size() = " << size());
425 return true;
426 }

429 /** Process the read reservations that are ready to be released in the
430 reservation queue
431 */
432 void move_read_reservation_forward() {
433 // Lock the pipe to avoid nuisance
434 std::unique_lock<std::mutex> lock { cb_mutex };
435
436 for (;;) {
437 if (r_rid_q.empty())
438 // No pending reservation, so nothing to do
439 break;
440 if (!r_rid_q.front().ready)
441 /* If the first reservation is not ready to be released, stop
442 because it is blocking all the following in the queue
443 anyway */
444 break;
445 // Remove the reservation to be released from the queue
446 r_rid_q.pop_front();
447 std::size_t n_to_pop;
448 if (r_rid_q.empty())
449 // If it was the last one, remove all the reservation
450 n_to_pop = read_reserved_frozen;
451 else
452 // Else remove everything up to the next reservation
453 n_to_pop = r_rid_q.front().start - cb.begin();
454 // No longer take into account these reserved slots
455 read_reserved_frozen -= n_to_pop;
456 // Release the elements from the FIFO
457 while (n_to_pop--)
458 cb.pop_front();
459 // Notify the clients waiting for some room to write in the pipe
460 read_done.notify_all();
461 /* ...and process the next reservation to see if it is ready to
462 be released too */
463 }
464 }

467 /** Process the write reservations that are ready to be released in the
468 reservation queue
469 */
470 void move_write_reservation_forward() {
471 // Lock the pipe to avoid nuisance
472 std::lock_guard<std::mutex> lg { cb_mutex };
473
474 for (;;) {
475 if (w_rid_q.empty())
476 // No pending reservation, so nothing to do
477 break;
478 // Get the first reservation
479 const auto &rid = w_rid_q.front();
480 if (!rid.ready)
481 /* If the reservation is not ready to be released, stop
482 because it is blocking all the following in the queue
483 anyway */
484 break;
485 // Remove the reservation to be released from the queue
486 w_rid_q.pop_front();
487 // Notify the clients waiting to read something from the pipe
488 write_done.notify_all();
489 /* ...and process the next reservation to see if it is ready to
490 be released too */
491 }
492 }
