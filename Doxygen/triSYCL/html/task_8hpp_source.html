82 /// Add a new task to the task graph and schedule for execution
83 void schedule(std::function<void(void)> f) {
84 /* To keep a copy of the task shared_ptr after the end of the
85 command group, capture it by copy in the following lambda.
86 */
87 auto execution = [f = std::move(f), task = shared_from_this()] {
88 // Wait for the required tasks to be ready
89 task->wait_for_producers();
90 task->prelude();
91 TRISYCL_DUMP_T("Execute the kernel");
92 // Execute the kernel
93 f();
94 task->postlude();
95 // Release the buffers that have been written by this task
96 task->release_buffers();
97 // Notify the waiting tasks that we are done
98 task->notify_consumers();
99 // Notify the queue we are done
100 task->owner_queue->kernel_end();
101 TRISYCL_DUMP_T("Task thread exit");
102 };
103 /* Notify the queue that there is a kernel submitted to the
104 queue. Do not do it in the task contructor so that we can deal
105 with command group without kernel and if we put it inside the
106 thread, the queue may have finished before the thread is
107 scheduled */
108 owner_queue->kernel_start();
109 /* \todo it may be implementable with packaged_task that would
110 deal with exceptions in kernels
111 */
112#ifndef TRISYCL_NO_ASYNC
113 /* If in asynchronous execution mode, execute the functor in a new
114 thread */
115 std::thread thread(execution);
116 TRISYCL_DUMP_T("Task thread started");
117 /** Detach the thread since it will synchronize by its own means
118
119 \todo This is an issue if there is an exception in the kernel
120 */
121 thread.detach();
122#else
123 // Just a synchronous execution otherwise
124 execution();
125#endif
126 }

129 /// Wait for the required producer tasks to be ready
130 void wait_for_producers() {
131 TRISYCL_DUMP_T("Task " << this << " waits for the producer tasks");
132 for (auto &t : producer_tasks)
133 t->wait();
134 // We can let the producers rest in peace
135 producer_tasks.clear();
136 }
137
138
139 /// Release the buffers that have been used by this task
140 void release_buffers() {
141 TRISYCL_DUMP_T("Task " << this << " releases the written buffers");
142 for (auto b: buffers_in_use)
143 b->release();
144 buffers_in_use.clear();
145 }
146
147
148 /// Notify the waiting tasks that we are done
149 void notify_consumers() {
150 TRISYCL_DUMP_T("Notify all the task waiting for this task " << this);
151 {
152 std::unique_lock<std::mutex> ul { ready_mutex };
153 execution_ended = true;
154 }
155 /* \todo Verify that the memory model with the notify does not
156 require some fence or atomic */
157 ready.notify_all();
158 }
159
160
161 /** Wait for this task to be ready
162
163 This is to be called from another thread
164 */
165 void wait() {
166 TRISYCL_DUMP_T("The task wait for task " << this << " to end");
167 std::unique_lock<std::mutex> ul { ready_mutex };
168 ready.wait(ul, [&] { return execution_ended; });
169 }

172 /** Register a buffer to this task
173
174 This is how the dependency graph is incrementally built.
175 */
176 void add_buffer(std::shared_ptr<detail::buffer_base> &buf,
177 bool is_write_mode) {
178 TRISYCL_DUMP_T("Add buffer " << buf << " in task " << this);
179 /* Keep track of the use of the buffer to notify its release at
180 the end of the execution */
181 buffers_in_use.push_back(buf);
182 // To be sure the buffer does not disappear before the kernel can run
183 buf->use();
184
185 std::shared_ptr<detail::task> latest_producer;
186 if (is_write_mode) {
187 /* Set this task as the latest producer of the buffer so that
188 another kernel may wait on this task */
189 latest_producer = buf->set_latest_producer(shared_from_this());
190 }
191 else
192 latest_producer = buf->get_latest_producer();
193
194 /* If the buffer is to be produced by a task, add the task in the
195 producer list to wait on it before running the task core
196
197 If a buffer is accessed first in write mode and then in read mode,
198 the task will add itself as a producer and will wait for itself
199 when calling \c wait_for_producers, we avoid this by checking that
200 \c latest_producer is not \c this
201 */
202 if (latest_producer && latest_producer != shared_from_this())
203 producer_tasks.push_back(latest_producer);
204 }

207 /// Execute the prologues
208 void prelude() {
209 TRISYCL_DUMP_T("task::prelude");
210
211 for (const auto &p : prologues)
212 p();
213 /* Free the functors that may own an accessor owning a buffer
214 preventing the command group to complete */
215 prologues.clear();
216 }
217
218
219 /// Execute the epilogues
220 void postlude() {
221 for (const auto &p : epilogues)
222 p();
223 /* Free the functors that may own an accessor owning a buffer
224 preventing the command group to complete */
225 epilogues.clear();
226 }
