38 /** Use the accessor with integers à la [i1][i2][i3] or C++23 [i1, i2,...]
39
40 \return decltype(auto) to return either a reference to the final
41 element when the indexing has been fully resolved or a proxy
42 object to handle the remaining [] */
43 template <std::integral... T> decltype(auto) operator[](T... indices) {
44 if constexpr (sizeof...(T) == 1)
