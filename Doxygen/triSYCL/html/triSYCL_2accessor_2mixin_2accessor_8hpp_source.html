25/// SYCL accessor mixin providing multi-dimensional access features
26template <typename T, int Dimensions> class accessor {
27
28 public:
29 /** Extension to SYCL: provide pieces of STL container interface
30 from mdspan */
31 using element_type = T;
32 /** Even if the buffer is read-only use a non-const type so at least
33 the current implementation can copy the data too */
34 using value_type = std::remove_cv_t<element_type>;
35
36 // Useful to handle buffers initialized from const values
37 using non_const_pointer = value_type*;
38
39 /** Get the number of dimensions of the buffer
40
41 Name inspired from ISO C++ P0009 mdspan papers
42 */
43 static auto constexpr rank() { return Dimensions; }
44
45 protected:
46 /// The memory lay-out of a buffer is a dynamic multidimensional array
47 using mdspan =
48 std::mdspan<element_type, std::dextents<std::size_t, Dimensions>>;
49
50 /** This is the multi-dimensional interface to the data that may point
51 to either allocation in the case of storage managed by SYCL itself
52 or to some other memory location in the case of host memory or
53 storage<> abstraction use
54 */
55 mdspan access;
56
57 /** Cast a SYCL range/id-like into a mdspan index array, which is an
58 array of std::size_t into an array of std::ptrdiff_t
59 */
60 template <typename BasicType, typename FinalType>
61 const std::array<typename mdspan::size_type, rank()>&
62 extents_cast(const detail::small_array<BasicType, FinalType, rank()>& sa) {
63 return reinterpret_cast<
64 const std::array<typename mdspan::size_type, rank()>&>(sa);
65 }
66
67 /// Set later the mdspan associated to this accessor
68 void set_access(const mdspan& a) { access = a; }
69
70 public:
71 /// Pointer type to element
72 using pointer = typename mdspan::accessor_type::data_handle_type;
73
74 /// Pointer type to const element
75 using const_pointer = const element_type*;
76
77 /// Reference type to the elements
78 using reference = typename mdspan::reference;
79
80 /// Used by the local accessor hack on top of host accessor
81 accessor() = default;
82
83 /// Create an accessor of dimensions r on top of data storage
84 accessor(pointer data, const range<rank()>& r)
85 : access { data, extents_cast(r) } {}
86
87 /// Create an accessor from another mdspan
88 accessor(const mdspan& m)
89 : access { m } {}
90
91 /// Update the accessor to target somewhere else
92 void update(pointer data, const range<rank()>& r) {
93 access = mdspan { data, extents_cast(r) };
94 }
95
96 /** Return a range object representing the size of the buffer in
97 terms of number of elements in each dimension as passed to the
98 constructor
99
100 \todo Cache it since it is const?
101 */
102 auto get_range() const {
103 range<Dimensions> r;
104 for (std::size_t i = 0; i < Dimensions; ++i)
105 r[i] = access.extent(i);
106 return r;
107 }
108
109 /** Returns the total number of elements in the buffer
110
111 Equal to get_range()[0] * ... * get_range()[Dimensions-1].
112
113 \todo Move these kinds of functions into a mixin between buffers
114 and accessors?
115
116 \todo Cache it since it is const?
117 */
118 std::size_t get_count() const {
119 return access.mapping().required_span_size();
120 }
121
122 /** Returns the size of the buffer storage in bytes
123
124 \todo rename to something else. In
125 http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2015/p0122r0.pdf
126 it is named bytes() for example
127
128 \todo Cache it since it is const?
129 */
130 std::size_t get_size() const { return get_count() * sizeof(value_type); }
131
132 /// Get the underlying storage
133 auto data() { return access.data_handle(); }
134
135 /** Access to an mdspan element with indices implementing a tuple
136 interface
137
138 \param[inout] some_mdspan is the mdspan to access
139
140 \param[in] tuple_like_indices are the indices to use
141
142 \return a reference to the element
143 */
144 static decltype(auto)
145 tuple_indexed_mdspan_access(auto&& some_mdspan,
146 const auto& tuple_like_indices) {
147 // Otherwise use the C++23 mdspan[i1, i2,...] indexing syntax
148 return std::invoke(
149 [&](auto... i) -> decltype(auto) { return some_mdspan[i...]; },
150 tuple_like_indices);
151 }
152
153 /** Access to an element with indices implementing a tuple
154 interface
155
156 \param[in] tuple_like_indices are the indices to use
157
158 \return a reference to the accessor element
159 */
160 decltype(auto) tuple_indexed_access(const auto& tuple_like_indices) {
161 return tuple_indexed_mdspan_access(access, tuple_like_indices);
162 }
163
164 /** Proxy object to transform an expression like
165 accessor[i1][i2][i3] into the implementation mdpsan(i1,i2,i3)
166 one index at a time.
167
168 It gathers intermediate [index] to finally call the mdspan
169 indexing operator once they are all available
170
171 \parameter[in] N is the number of indices which can be stored in
172 this proxy
173 */
174 template <std::size_t N> struct track_index {
175 /// Keep a reference to the mdspan to eventually resolve the
176 /// indexing
177 mdspan& mds;
178
179 /// The list of indices in the order of [i1][i2][i3]...
180 std::array<std::size_t, N> indices;
181
182 /// Construct the initial tracking object from the accessor
183 track_index(accessor& a)
184 : mds { a.access } {}
185
186 /// Create a tracking object from an mdspan and a list of indices
187 template <typename... Index>
188 track_index(mdspan& m, Index&&... inds)
189 : mds { m }
190 // Typically there will be N - 1 indices in the array of size N
191 , indices { std::forward<Index>(inds)... } {}
192
193 /// The individual indexing operator
194 decltype(auto) operator[](std::size_t index) {
195 // Keep track of the new index in the last element
196 indices[N - 1] = index;
197 if constexpr (N == accessor::rank())
198 // If we have accumulated all the indices, call the mdspan
199 // indexing function
200 return tuple_indexed_mdspan_access(mds, indices);
201 else
202 // Otherwise return a tracker from the current indices and
203 // with 1 more room for the next index
204 return std::make_from_tuple<track_index<N + 1>>(
205 std::tuple_cat(std::forward_as_tuple(mds), indices));
206 }
207 };
208};
