40/** The buffer accessor abstracts the way buffer data are accessed
41 inside a kernel in a multidimensional variable length array way.
42
43 This implementation relies on std::mdspan to provide this nice
44 syntax and behavior.
45
46 Right now the aim of this class is just to access to the buffer in
47 a read-write mode, even if capturing the accessor from a lambda
48 make it const (since in examples we have lambda with [=] without
49 mutable lambda).
50
51 \todo Use the access::mode
52*/
53template <typename T, int Dimensions, access::mode Mode,
54 access::target Target /* = access::global_buffer */>
55class accessor
56 : public detail::accessor_base
57 , public facade::accessor<mixin::accessor<T, Dimensions>>
58 , public std::enable_shared_from_this<accessor<T, Dimensions, Mode, Target>>
59 , public detail::debug<accessor<T, Dimensions, Mode, Target>> {
60 /** Keep a reference to the accessed buffer
61
62 Beware that it owns the buffer, which means that the accessor
63 has to be destroyed to release the buffer and potentially
64 unblock a kernel at the end of its execution
65 */
66 std::shared_ptr<detail::buffer<T, Dimensions>> buf;
67
68 /// Where most of the user-facing interface dwells
69 using facade = facade::accessor<mixin::accessor<T, Dimensions>>;
70
71 public:
72#ifdef TODO

78 using reverse_iterator = typename array_view_type::reverse_iterator;

84 accessor() = default;

86 /** Construct a host accessor from an existing buffer
87
88 \todo fix the specification to rename target that shadows
89 template parm
90 */
91 accessor(std::shared_ptr<detail::buffer<T, Dimensions>> target_buffer)
92 : facade { target_buffer->access }
93 , buf { target_buffer } {
94 target_buffer->template track_access_mode<Mode>();
95 TRISYCL_DUMP_T("Create a host accessor write = " << is_write_access());
96 static_assert(Target == access::target::host_buffer,
97 "without a handler, access target should be host_buffer");
98 /* The host needs to wait for all the producers of the buffer to
99 have finished */
100 buf->wait();
101
102#ifdef TRISYCL_OPENCL
103 /* For the host context, we are obligated to update the buffer state
104 during the accessors creation, otherwise we have no way of knowing
105 if a buffer was modified on the host. This is only true because
106 host accessors are blocking
107 */
108 trisycl::context ctx;
109 buf->update_buffer_state(ctx, Mode, facade::get_size(), facade::data());
110#endif
111 }

113 /** Construct a device accessor from an existing buffer
114
115 \todo fix the specification to rename target that shadows
116 template parm
117 */
118 accessor(std::shared_ptr<detail::buffer<T, Dimensions>> target_buffer,
119 handler& command_group_handler)
120 : facade { target_buffer->access }
121 , buf { target_buffer } {
122 target_buffer->template track_access_mode<Mode>();
123 TRISYCL_DUMP_T("Create a kernel accessor write = " << is_write_access());
124 static_assert(Target == access::target::global_buffer ||
125 Target == access::target::constant_buffer,
126 "access target should be global_buffer or constant_buffer "
127 "when a handler is used");
128 // Register the buffer to the task dependencies
129 task = buffer_add_to_task(buf, &command_group_handler, is_write_access());
130 }
131
132 /** Register the accessor once a \c std::shared_ptr is created on it
133
134 This is to be called from outside once the object is created. It
135 has been tried directly inside the contructor, but calling \c
136 shared_from_this() from the constructor dead-lock with
137 libstdc++6
138
139 \todo Double-check with the C++ committee on this issue.
140 */
141 void register_accessor() {
142 if (!task->get_queue()->is_host()) {
143 // To keep alive this accessor in the following lambdas
144 auto acc = this->shared_from_this();
145 // Attach the accessor to the task and get its order
146 set_order(task->register_accessor(acc));
147#ifdef TRISYCL_OPENCL
148 /* Before running the kernel, make sure the cl_mem behind this
149 accessor is up-to-date on the device if needed and pass it to
150 the kernel */
151 task->add_prelude([=] { acc->copy_in_cl_buffer(); });
152 // After running the kernel, deal with some copy-back if needed
153 task->add_postlude([=] {
154 /* Even if this function does nothing, it is required to
155 have the capture of acc to keep the accessor alive across
156 the kernel execution up to the execution postlude */
157 acc->copy_back_cl_buffer();
158 });
159#endif
160 }
161 }
162
163 /// Get the buffer used to create the accessor
164 detail::buffer<T, Dimensions>& get_buffer() { return *buf; }

166 /** Test if the accessor has a read access right
167
168 \todo Strangely, it is not really constexpr because it is not a

174 constexpr bool is_read_access() const {
175 return Mode == access::mode::read || Mode == access::mode::read_write ||
176 Mode == access::mode::discard_read_write;

187 constexpr bool is_write_access() const {
188 return Mode == access::mode::write || Mode == access::mode::read_write ||
189 Mode == access::mode::discard_write ||
190 Mode == access::mode::discard_read_write;

209 /* Create the OpenCL buffer and copy in it the data from the host if

211 */

213 buf->update_buffer_state(ctx, Mode, facade::get_size(), facade::data());
214 }

224 }
