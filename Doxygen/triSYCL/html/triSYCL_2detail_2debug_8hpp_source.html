130/** Wrap a kernel functor in some tracing messages to have start/stop
131 information when TRISYCL_TRACE_KERNEL macro is defined */
132template <typename KernelName, typename Functor>
133auto trace_kernel(Functor f) {
134#ifdef TRISYCL_TRACE_KERNEL
135 // Inject tracing message around the kernel
136 return [=] () mutable {
137 /* Since the class KernelName may just be declared and not really
138 defined, just use it through a class pointer to have
139 typeid().name() not complaining */
140 TRISYCL_INTERNAL_DUMP(
141 "Kernel started "
142 << boost::typeindex::type_id<KernelName *>().pretty_name());
143 f();
144 TRISYCL_INTERNAL_DUMP(
145 "Kernel stopped "
146 << boost::typeindex::type_id<KernelName *>().pretty_name());
147 };
148#else
149 // Identity by default
150 return f;
151#endif
152}
