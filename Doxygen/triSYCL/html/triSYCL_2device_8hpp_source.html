86 /** Construct a device class instance using the device selector
87 provided
88
89 Return errors via C++ exception class.
90
91 \todo Make it non-explicit in the specification?
92 */
93 explicit device(const device_selector &ds) {
94 auto devices = device::get_devices();
95 if (devices.empty())
96 // \todo Put a SYCL exception
97 throw std::domain_error("No device at all! Internal error...");
98
99 /* Find the device with the best score according to the given
100 device_selector */
101 auto max = std::max_element(devices.cbegin(), devices.cend(),
102 [&] (const device &d1, const device &d2) {
103 return ds(d1) < ds(d2);
104 });
105 if (ds(*max) < 0)
106 // \todo Put a SYCL exception
107 throw std::domain_error("No device selected because no positive "
108 "device_selector score found");
109
110 // Create the current device as a shared copy of the selected one
111 implementation = max->implementation;
112 }
