26/** Some implementation details about the SYCL queue
27 */
28struct queue : detail::debug<detail::queue> {
29 /// Track the number of kernels still running to wait for their completion
30 std::atomic<size_t> running_kernels;
31
32 /// To signal when all the kernels have completed
33 std::condition_variable finished;
34 /// To protect the access to the condition variable
35 std::mutex finished_mutex;
36
37
38 /// Initialize the queue with 0 running kernel
39 queue() : running_kernels { 0 } {}
40
41 /// Wait for all kernel completion
42 void wait_for_kernel_execution() {
43 TRISYCL_DUMP_T("Queue waiting for kernel completion");
44 std::unique_lock<std::mutex> ul { finished_mutex };
45 finished.wait(ul, [&] {
46 // When there is no kernel running in this queue, we are ready to go
47 return running_kernels == 0;
48 });
49 }
50
51
52 /// Signal that a new kernel started on this queue
53 void kernel_start() {
54 TRISYCL_DUMP_T("A kernel has been added to the queue");
55 // One more kernel
56 ++running_kernels;
57 }
58
59
60 /// Signal that a new kernel finished on this queue
61 void kernel_end() {
62 TRISYCL_DUMP_T("A kernel of the queue ended");
63 std::unique_lock<std::mutex> ul { finished_mutex };
64 if (--running_kernels == 0) {
65 // Micro-optimization: unlock before the notification
66 // https://en.cppreference.com/w/cpp/thread/condition_variable/notify_all
67 ul.unlock();
68 /* It was the last kernel running, so signal the queue just in
69 case it was working for it for completion
70
71 In some cases several threads might want to wait for the
72 same queue, because of this \c notify_one is not be enough
73 and a \c notify_all is needed
74 */
75 finished.notify_all();
76 }
77 }
