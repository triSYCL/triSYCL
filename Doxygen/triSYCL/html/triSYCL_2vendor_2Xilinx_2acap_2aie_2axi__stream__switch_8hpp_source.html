165 /// Create a router minion on this switch using a fiber executor
166 router_minion(axi_stream_switch &axi_ss,
167 ::trisycl::detail::fiber_pool &fiber_executor,
168 int capacity = axi_ss_geo::latency)
169 : capacity { capacity }
170 , axi_ss { axi_ss }
171 {
172 /// Use a fiber as a data mover from the input queue
173 futures.push_back(fiber_executor.submit([&] {
174 TRISYCL_DUMP_T("router_minion " << this << " on tile("
175 << router_minion::axi_ss.x_coordinate
176 << ',' << router_minion::axi_ss.y_coordinate
177 << ") on fiber " << boost::this_fiber::get_id()
178 << " starting with buffered_channel " << &c);
179 for (;;) {
180 TRISYCL_DUMP_T("router_minion " << this << " on tile("
181 << router_minion::axi_ss.x_coordinate
182 << ',' << router_minion::axi_ss.y_coordinate
183 << ") on fiber " << boost::this_fiber::get_id()
184 << " reading from buffered_channel "
185 << &c << " ...");
186 auto v = c.value_pop();
187 if (v.shutdown_request)
188 // End the execution on shutdown packet reception
189 break;
190 TRISYCL_DUMP_T("router_minion " << this << " on tile("
191 << router_minion::axi_ss.x_coordinate
192 << ',' << router_minion::axi_ss.y_coordinate
193 << ") on fiber " << boost::this_fiber::get_id()
194 << " routing data value " << v.data
195 << " received from buffered_channel " << &c);
196 // The routing itself is a blocking write on each output
197 for (auto &o : outputs) {
198 TRISYCL_DUMP_T("router_minion " << this << " on tile("
199 << router_minion::axi_ss.x_coordinate
200 << ',' << router_minion::axi_ss.y_coordinate
201 << ") on fiber " << boost::this_fiber::get_id()
202 << " forwarding to router_minion " << o.get());
203 o->write(v);
204 }
205 }
206 TRISYCL_DUMP_T("router_minion " << this << " on tile("
207 << router_minion::axi_ss.x_coordinate
208 << ',' << router_minion::axi_ss.y_coordinate
209 << ") on fiber " << boost::this_fiber::get_id()
210 << " shutting down");
211 }));

243 /** Validate the user port number and translate it to the physical
244 port number
245
246 \param[in] user_port is the logical port number, starting to 0
247
248 \param[in] physical_port_min is the physical lower port number to map to
249
250 \param[in] physical_port_max is the physical higher port number
251 to map to
252
253 \throws ::trisycl::runtime_error if the port number is invalid
254
255 \return the physical port number in the switch corresponding to
256 the logical port
257
258 \todo This makes sense only for core tile
259 */
260 static auto inline translate_port = [] (int user_port,
261 auto physical_port_min,
262 auto physical_port_max,
263 const auto& error_message) {
264 // Cast to int since the physical port might be the enum types
265 auto port_min = static_cast<int>(physical_port_min);
266 auto port_max = static_cast<int>(physical_port_max);
267 auto last_user_port = port_max - port_min;
268 if (user_port < 0 || user_port > last_user_port)
269 throw ::trisycl::runtime_error {
270 (boost::format {
271 "%1%: %2% is not a valid port number between 0 and %3%" }
272 % error_message % user_port % last_user_port).str() };
273 // Return the value as the requested enum port
274 return static_cast<decltype(physical_port_min)>(port_min + user_port);
275 };

419 /// Start the tile infrastructure associated to the AIE device
420 void start(int x, int y, ::trisycl::detail::fiber_pool &fiber_executor) {
421 x_coordinate = x;
422 y_coordinate = y;
423 // Add a router worker on all switch input
424 for (auto& p : input_ports) {
425 auto port_enum =
426 magic_enum::enum_cast<spl>(std::distance(input_ports.begin(), &p));
427#ifdef TRISYCL_DEBUG
428 auto port_name = magic_enum::enum_name(port_enum.value());
429#endif
430 TRISYCL_DUMP_T("axi_stream_switch " << this
431 << " (" << x_coordinate << ',' << y_coordinate
432 << ") start initialize input: " << port_name);
433 // If the port comes from the FIFO, use a FIFO-featured
434 // router_minion instead
435 if (ranges::find(axi_ss_geo::s_fifo_range, port_enum)
436 != std::end(axi_ss_geo::s_fifo_range))
437 /* A special AXI switch routing interface which provides a
438 FIFO. Since a router_minion has already some internal FIFO
439 behavior, some deeper FIFO behavior can actually be
440 implemented with a specific router_minion with a deeper
441 FIFO internally, to avoid having 2 active components, the
442 FIFO + the router_minion */
443 p = std::make_shared<router_minion>(*this,
444 fiber_executor,
445 axi_ss_geo::fifo_depth);
446 else
447 p = std::make_shared<router_minion>(*this, fiber_executor);
448 }
449 // Since the FIFO behavior is actually implemented with a
450 // router_minion with a deeper FIFO internally, the "input" of the
451 // "FIFO" can be just directly connected to this specific
452 // router_minion
453 for (auto [o, i] : ranges::views::zip(axi_ss_geo::m_fifo_range,
454 axi_ss_geo::s_fifo_range))
455 output(o) = input(i);
456 }

459 /** Compute the edge angle for a port name according a list of
460 description associating a port name prefix and an associated
461 angle in degree
462 */
463 int
464 tikz_edge_angle(const std::string_view& port_name,
465 const std::initializer_list<std::tuple<std::string_view,
466 int>>& desc) const {
467 for (auto& [prefix, angle] : desc)
468 if (port_name.starts_with(prefix))
469 return angle;
470 assert(0 && "Unknown port name");
471 return 0;
472 }

475 /// Display the internal switch connections with routing arrows
476 /// perpendicular to the borders
477 void display_routing_configuration(latex::context& c) const {
478 for (const auto& [i, p] : ranges::views::enumerate(input_ports)) {
479 auto input_port = static_cast<spl>(i);
480 auto rm = get_as_router_minion(p.get());
481 for (auto output_port : rm->get_master_port_dests()) {
482 auto input_port_name = magic_enum::enum_name(input_port);
483 // Compute the angle of the starting arrow from the input
484 auto input_angle = tikz_edge_angle(input_port_name,
485 { { "east", 180 }, ///< To the left
486 { "north", -90 }, ///< Downwards
487 { "west", 0 }, ///< To the right
488 { "dma", 90 }, ///< Upwards
489 { "fifo", 90 },
490 { "me", 90 },
491 { "tile_ctrl", 90 },
492 { "south", 90 } });
493 auto output_port_name = magic_enum::enum_name(output_port);
494 // Compute the angle of the ending arrow to the output
495 auto output_angle = tikz_edge_angle(output_port_name,
496 { { "east", 180 }, ///< To the left
497 { "north", -90 }, ///< Downwards
498 { "dma", 0 }, ///< To the right
499 { "fifo", 0 },
500 { "me", 0 },
501 { "tile_ctrl", 0 },
502 { "west", 0 },
503 { "south", 90 } }); ///< Upwards
504 // Use specific color for core emission or reception
505 auto color = input_port_name.starts_with("me") ? "red"
506 : output_port_name.starts_with("me") ? "teal"
507 : input_port_name.starts_with("dma") ? "magenta"
508 : output_port_name.starts_with("dma") ? "violet" : "blue";
509 c.add((boost::format { R"(
510 \draw[%5%] (node cs:name=S%1%)
511 to [out=%2%, in=%4%] (node cs:name=M%3%);)" }
512 % c.clean_node(input_port_name) % input_angle
513 % c.clean_node(output_port_name) % output_angle
514 % color).str());
515 }
516 }
517 }

520 /// Display a border of the AXI stream switch
521 template <typename FG, typename FL, typename... EnumRanges>
522 auto display_border(const std::string_view& node_attribute,
523 FG&& global_coordinate_function,
524 FL&& local_coordinate_function,
525 EnumRanges&&... enum_ranges) const {
526 std::string out;
527 // Index each connection displayed in order
528 auto i = 0;
529 boost::hana::for_each(boost::hana::make_tuple(enum_ranges...),
530 [&] (auto&& r) {
531 for (auto p : r) {
532 auto [x, y] = local_coordinate_function(i);
533 out += (boost::format { R"(
534 \coordinate(%3%) at %2%;
535 \node[%1%] at %2% {%4%};)" }
536 % node_attribute
537 % global_coordinate_function(x, y)
538 // Make a node name from enum name prefixed with master/slave statu7s
539 % latex::context::clean_node(magic_enum::enum_name(p),
540 is_axi_master(p) ? "M" : "S")
541 // Use the enum name as the port name to display
542 % magic_enum::enum_name(p)).str();
543 ++i;
544 };
545 });
546 return out;
547 }
