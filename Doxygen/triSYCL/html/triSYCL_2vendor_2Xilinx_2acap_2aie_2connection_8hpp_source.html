113/// A FIFO channel connection
114/// \todo factorize out router minion
115class fifo_channel : public communicator_port {
116 /// FIFO capacity queue
117 /// \todo check with hardware team for the value
118 auto static constexpr capacity = 16;
119
120 /// Payload data type
121 using value_type = axi_packet::value_type;
122
123 /// The underlying FIFO
124 boost::fibers::buffered_channel<axi_packet> c { capacity };
125
126public:
127
128 /// Enqueue a packet to the FIFO channel
129 void write(const axi_packet &v) override {
130 c.push(v);
131 }
132
133
134 /** Try to enqueue a packet to the FIFO channel
135
136 \return true if the packet is correctly enqueued
137 */
138 bool try_write(const axi_packet &v) override {
139 return c.try_push(v) == boost::fibers::channel_op_status::success;
140 }
141
142
143 /// Waiting read from the FIFO channel
144 value_type read() override {
145 return c.value_pop().data;
146 }
147
148
149 /** Non-blocking read from the FIFO channel
150
151 \return true if the value was correctly read
152 */
153 bool try_read(value_type &v) override {
154 axi_packet p;
155 if (c.try_pop(p) == boost::fibers::channel_op_status::success) {
156 v = p.data;
157 return true;
158 }
159 return false;
160 }

162};

259/// A connection between some AXI Stream Switch
260struct connection {

264 /** Create a connection
265
266 \param[in] pipe is the communication media to be used
267 */
268 template <typename Pipe>
269 connection(Pipe && pipe)
270 : p { new std::any { std::forward<Pipe>(pipe) } } {}
271
272
273 /// Get the input port of this connection
274 auto in() {
275 return input { *this };

277
278
279 /// Get the output port of this connection
280 auto out() {
281 return output { *this };

285 /// Base class of connection port
286 struct base {
287 /// Keep track of the communication media
288 std::shared_ptr<std::any> p;
289
290
291 /// Construct a port from a connection
292 base(const connection &c)
293 : p { c.p } {}
294
295 base() = default;
296
297
298 /** Access the type erased pipe as type T
299
300 \param[in] T is the type of the data transmitted by the
301 connection
302
303 \throws trisycl::runtime_error if the connection has not the
304 correct type
305 */
306 template <typename T>
307 auto pipe_of() {
308 try {
309 return std::any_cast<::trisycl::sycl_2_2::static_pipe<T, 4>>(*p);
310 } catch (std::bad_any_cast &) {
311 throw ::trisycl::runtime_error {
312 "The current connection is not of type "s
313 + boost::typeindex::type_id<T>().pretty_name() };
314 }
315 }
316
317 };
318
319
320 /// Abstract an input port
321 struct input : base {
322 /// Recycle the base constructors
323 using base::base;
324
325 /** Get the input port
326
327 \param[in] InputT is the data type to be used in the transfers
328
329 \param[in] Target specifies if the connection is blocking or
330 not
331
332 \throws trisycl::runtime_error if the input is not connected
333 */
334 template <typename InputT, access::target Target>
335 auto in() {
336 if (!p)
337 throw ::trisycl::runtime_error { "This input is not connected" };
338 return
339 pipe_of<InputT>().template get_access<access::mode::read, Target>();
340 }
341 };
342
343
344 /// Abstract an output port
345 struct output : base {
346 /// Recycle the base constructors
347 using base::base;
348
349 /** Get the output port
350
351 \param[in] InputT is the data type to be used in the transfers
352
353 \param[in] Target specifies if the connection is blocking or
354 not
355
356 \throws trisycl::runtime_error if the output is not connected
357 */
358 template <typename OutputT, access::target Target>
359 auto out() {
360 if (!p)
361 throw ::trisycl::runtime_error { "This output is not connected" };
362 return
363 pipe_of<OutputT>().template get_access<access::mode::write, Target>();
364 }
365 };
