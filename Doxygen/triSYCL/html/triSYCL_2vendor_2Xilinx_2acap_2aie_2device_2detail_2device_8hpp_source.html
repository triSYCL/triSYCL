80 /** Access to the common infrastructure part of a tile
81

93 /** Apply a function for each tile index of the device
94
95 \param f is a callable that is to be called like \c f(x,y) for
96 each tile
97 */
98 template <typename F> void for_each_tile_index(F&& f) {
99 for (auto [x, y] :
100 ranges::views::cartesian_product(ranges::views::iota(0, geo::x_size),
101 ranges::views::iota(0, geo::y_size)))
102 f(x, y);
103 };

105 /** Apply a function for each tile infrastructure of the device
106
107 \param f is a callable that is to be called like \c f(x,y) for
108 each tile

110 template <typename F> void for_each_tile(F&& f) {
111 for_each_tile_index([&](auto x, auto y) { f(tile(x, y)); });

167 /** Connect the ports of 2 tiles or shims together with an
168 hyperspace switched circuit, jumping over the underlying routing
169 infrastructure.
170
171 This is used to speed up communication in emulation and to skip
172 the need of a routing planner.
173
174 \todo To deprecate and implement in another way?
175
176 \param[in] T is the type of the data to be transferred
177
178 \param[in] SrcPort is the type of the source port, such as
179 port::tile or port::shim
180
181 \param[in] DstPort is the type of the destination port, such as
182 port::tile or port::shim
183
184 \throws trisycl::runtime_error if some coordinates or port
185 numbers are invalid
186
187 \todo Refactor, make the difference between user & physical ports
188 */
189 template <typename SrcPort, typename DstPort>
190 void connect(SrcPort&& src, DstPort&& dst) {
191 static_assert(std::is_same_v<SrcPort, port::tile> ||
192 std::is_same_v<SrcPort, port::shim>,
193 "SrcPort type should be port::tile or port::shim");
194 static_assert(std::is_same_v<DstPort, port::tile> ||
195 std::is_same_v<DstPort, port::shim>,
196 "DstPort type should be port::tile or port::shim");
197 // Get the destination in the final switch
198 auto channel = [&] {
199 if constexpr (std::is_same_v<DstPort, port::tile>)
200 return tile(dst.x, dst.y).in_connection(dst.port);
201 else if constexpr (std::is_same_v<DstPort, port::shim>)
202 return shim(dst.x).bli_in_connection(dst.port);
203 }();
204 // And connect the source port to it
205 if constexpr (std::is_same_v<SrcPort, port::tile>)
206 tile(src.x, src.y).out_connection(src.port) = channel;
207 else if constexpr (std::is_same_v<SrcPort, port::shim>)
208 shim(src.x).bli_out_connection(src.port) = channel;
209 }
210
211 /// Apply a function on all the AXI stream of the neighborhood of each tile
212 template <typename F> void for_each_tile_neighborhood(F&& f) {
213 for_each_tile_index([&](auto x, auto y) {
214 // No CTAD yet with Boost::Hana and Clang++-10 (but works with g++-9)
215 boost::hana::for_each(
216 geo::core_axi_stream_switch::interconnect, [&](auto connections) {
217 auto [dx, dy, master_range, slave_range] = connections;
218 if (geo::is_x_y_valid(x + dx, y + dy))
219 for (auto [o, i] : ranges::views::zip(master_range, slave_range))
220 f(x, y, x + dx, y + dy, o, i);
221 });
222 });
223 }
224
225 /// Build some device level infrastructure
226 device() {
227 // Initialize all the tiles with their network connections first
228 for_each_tile_index([&](auto x, auto y) {
229 // Create & start the tile infrastructure
230 tile(x, y) = { x, y, *this, fiber_executor };
231 });
232 for_each_tile_x_index([&](auto x) {
233 // Start the shim infrastructure
234 shim(x).start(x, fiber_executor);
235 });
236
237 // Only then we can connect the inter-core tile AXI stream NoC
238 for_each_tile_neighborhood(
239 [&](auto x, auto y, auto nx, auto ny, auto m, auto s) {
240 tile(x, y).output(m) = tile(nx, ny).input(s);
241 });
242
243 /// Make the connections around the shim tiles
244 for_each_tile_x_index([&](auto x) {
245 // The connection from shim tile to core tile
246 for (auto [o, i] :
247 ranges::views::zip(sass::m_north_range, cass::s_south_range))
248 shim(x).output(o) = tile(x, 0).input(i);
249 // The connection from core tile to shim tile
250 for (auto [o, i] :
251 ranges::views::zip(cass::m_south_range, sass::s_north_range))
252 tile(x, 0).output(o) = shim(x).input(i);
253 if (!geo::is_east_column(x))
254 for (auto [o, i] :
255 ranges::views::zip(sass::m_east_range, sass::s_west_range))
256 shim(x).output(o) = shim(x + 1).input(i);
257 if (!geo::is_west_column(x))
258 for (auto [o, i] :
259 ranges::views::zip(sass::m_west_range, sass::s_east_range))
260 shim(x).output(o) = shim(x - 1).input(i);
261 });
262 }

264 /** Display the device layout
265
266 \param[in] file_name is the name of the file to write the LaTeX
267 drawing into or use std::cerr by default
268 */
269 auto display(const std::string& file_name = {}) {
270 // Compute the drawing size starting with the individual tile size
271 auto tile_size = tile(0, 0).display_size();
272 // And expanding according to the device size. Add 1 in each
273 // dimension to fit in the page because of the borders
274 auto x_size = tile_size.x() * geo::x_size + 1;
275 auto y_size = tile_size.y() * geo::y_size + 1;
276 /// The LaTeX generation is handled by a LaTeX context
277 latex::context c { { x_size, y_size }, file_name };
278
279 for_each_tile_index([&](auto x, auto y) { tile(x, y).display(c); });
280
281 // Connect each tile to its neighbors
282 for_each_tile_neighborhood(
283 [&](auto x, auto y, auto nx, auto ny, auto m, auto s) {
284 c.add((boost::format { R"(
285 \draw (node cs:name=TileX%1%Y%2%M%3%)
286 -- (node cs:name=TileX%4%Y%5%S%6%);)" } %
287 x % y % c.clean_node(magic_enum::enum_name(m)) % nx % ny %
288 c.clean_node(magic_enum::enum_name(s)))
289 .str());
290 });
291
292 c.display();
293 }
