35/** Implement a small DSL to chain DMA & locking operations like
36 allowed by the hardware
37
38 \param[in] Tile is the type of tile to keep a reference on it
39
40 \param[in] DMA is the type of the DMA involved in the operation
41
42 \param[in] LockRef is the type of the reference to the lock once a
43 lock() operation is involved. Otherwise just use a \c bool as a
44 dummy type internally.
45*/
46template <typename Tile, typename DMA, typename LockRef = bool> class dma_dsl {
47 /// Keep track of the tile involved in the operation chain
48 Tile& tile;
49
50 /// Keep track of the DMA involved in the operation chain
51 DMA& dma;
52
53 /** Keep track of the current lock involved in the operation chain
54 if any */
55 LockRef lck;
56
57 public:
58 /// Keep track of operation with only DMA involved
59 dma_dsl(Tile& t, DMA& d)
60 : tile { t }
61 , dma { d } {}
62
63 /// Keep track of operation with DMA and lock involved
64 dma_dsl(Tile& t, DMA& d, LockRef l)
65 : tile { t }
66 , dma { d }
67 , lck { l } {}
68
69 /// Select a lock for the following operations
70 auto lock(int l) {
71 auto& lock_ref = tile.mem().lock(l);
72 return dma_dsl<Tile, DMA, decltype(lock_ref)> { tile, dma, lock_ref };
73 }
74
75 /// Enqueue a DMA receive command
76 template <typename Cmd> auto& receive(Cmd&& cmd) {
77 dma.receive(std::forward<Cmd>(cmd));
78 // To be able to chain another DMA operation on it
79 return *this;
80 }
81
82 /// Enqueue a DMA transfer command
83 template <typename Cmd> auto& send(Cmd&& cmd) {
84 dma.send(std::forward<Cmd>(cmd));
85 // To be able to chain another DMA operation on it
86 return *this;
87 }
88
89 /** Wait for the transfers to complete
90
91 The API assumes that the program is written in such a way that
92 at some point the queue will be empty so there is no continuous
93 DMA command enqueuing for example. */
94 auto& wait() {
95 dma.wait();
96 // To be able to chain another DMA operation on it
97 return *this;
98 }
99};

101/// Generic DMA
102template <typename AXIStreamSwitch, typename SpecializedDMA>
103class dma : ::trisycl::detail::debug<SpecializedDMA> {
104
105 /// Use std::span for 1D contiguous transfers only for now
106 // using dma_command = std::span<axi_packet::value_type>;
107
108 private:
109 /// The type of commands executable by the DMA. Very generic for now
110 using dma_command = std::function<void(void)>;
111
112 /// Queue for generic DMA commands
113 boost::fibers::buffered_channel<dma_command> c { 8 };
114
115 /// To shepherd the fiber playing the data mover behind the DMA
116 ::trisycl::detail::fiber_pool::future<void> data_mover;
117
118 /// Count the in-flight commands
119 std::atomic<int> nb_command { 0 };
120
121 /// To wait for the DMA command queue to drain
122 boost::fibers::condition_variable waiting_room;
123
124 /// To protect the waiting mechanism
125 boost::fibers::mutex waiting_mutex;
126
127 /// Wait for the fiber to complete
128 void join() {
129 // Get the value of the future, to get an exception if any
130 data_mover.get();
131 }
132
133 /// Launch the data mover on the execution pool
134 template <typename Invocable>
135 void launch(::trisycl::detail::fiber_pool& fiber_executor, Invocable&& f) {}
136
137 /// Dequeue a DMA command
138 auto pop_command(auto& command) { return c.pop(command); }
139
140 /// Notify a command has been processed
141 void commit_command() {
142 if (--nb_command == 0)
143 // Warn anyone waiting for the DMA queue to drain it might be empty now
144 waiting_room.notify_all();
145 }
146
147 protected:
148 /// Enqueue a DMA command
149 template <typename Cmd> void push_command(Cmd&& command) {
150 ++nb_command;
151 c.push(std::forward<Cmd>(command));
152 }
153
154 public:
155 /// Start the DMA engine using an executor
156 dma(::trisycl::detail::fiber_pool& fe) {
157 data_mover = fe.submit([&] {
158 dma_command sp;
159 for (;;) {
160 // Read a DMA command
161 if (this->pop_command(sp) != boost::fibers::channel_op_status::success)
162 // The channel is closed, stop working
163 break;
164 // Execute the command
165 sp();
166 commit_command();
167 }
168 });
169 }
170
171 /// Destructor handling the correct infrastructure shutdown
172 ~dma() {
173 // Close the command queue so the DMA does not accept any work
174 c.close();
175 /* Wait for the commands to drain to avoid calling
176 std::terminate on destruction... */
177 join();
178 }
179
180 /** Wait for the transfers to complete
181
182 The API assumes that the program is written in such a way that
183 at some point the queue will be empty so there is no continuous
184 DMA command enqueuing for example. */
185 SpecializedDMA& wait() {
186 if (nb_command != 0) {
187 // We need to wait because there are still operations executing
188 std::unique_lock lk { waiting_mutex };
189 waiting_room.wait(lk, [&] { return nb_command == 0; });
190 }
191 /* To be able to chain another DMA operation on it, return the DMA
192 object itself */
193 return static_cast<SpecializedDMA&>(*this);
194 }
195};

202class receiving_dma
203 : public dma<AXIStreamSwitch, receiving_dma<AXIStreamSwitch>>
204 , public communicator_port {
205 using dma_base = dma<AXIStreamSwitch, receiving_dma<AXIStreamSwitch>>;
206
207 /* boost::fibers::unbuffered_channel has no try_push() function, so
208 use a buffered version for now
209
210 \todo open a GitHub issue on Boost.Fiber
211 */
212 boost::fibers::buffered_channel<axi_packet> fifo { 8 };
213
214 /// Keep track of the AXI stream switch owning this port for debugging
215 AXIStreamSwitch& axi_ss;
216
217 public:
218 /// Start the receiving DMA engine using an executor
219 receiving_dma(AXIStreamSwitch& axi_ss, ::trisycl::detail::fiber_pool& fe)
220 : dma_base { fe }
221 , axi_ss { axi_ss } {}
222
223 /** Enqueue a DMA transfer to receive a span
224
225 Use std::span for 1D contiguous transfers only for now
226 */
227 void receive(std::span<axi_packet::value_type> sp) {
228 this->push_command([=, this] {
229 /* Write each element received from input router port into the
230 memory described by DMA operation */
231 for (auto& e : sp) {
232 e = read();
233 }
234 });
235 }
236
237 /// The network sends some data to the DMA receiver input
238 void write(const axi_packet& v) override { fifo.push(v); }
239
240 /** The network try to send some data to the DMA receiver input
241
242 \return true if the packet is correctly enqueued or if the
243 stream is closed, in order to to move on since we are in the
244 tear-down phase of the infrastructure
245
246 \todo think about it again...
247 */
248 bool try_write(const axi_packet& v) override {
249 return this->fifo.try_push(v) != boost::fibers::channel_op_status::full;
250 }
251
252 /// Waiting read by a tile program on a core input port from the switch
253 axi_packet::value_type read() override { return fifo.value_pop().data; }
254
255 /** Non-blocking read to a core input port
256
257 \return true if the value was correctly read
258 */
259 bool try_read(axi_packet::value_type& v) override {
260 axi_packet p;
261
262 if (fifo.try_pop(p) == boost::fibers::channel_op_status::success) {
263 v = p.data;
264 return true;
265 }
266 return false;
267 }
268};
