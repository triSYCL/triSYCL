27/** Some geographic information about the AI Engine array
28
29 \param Layout is a class describing the basic layout of the CGRA,
30 setting for example the x & y size
31*/
32template <typename Layout>
33struct geography : Layout {
34 /// Keep the layout used for this CGRA
35 using layout = Layout;
36
37 /// Minimum horizontal coordinate
38 static auto constexpr x_min = 0;
39
40 /// Minimum vertical coordinate
41 static auto constexpr y_min = 0;
42
43 /// Number of tile horizontally in the CGRA
44 static auto constexpr x_size = layout::x_max + 1;
45
46 /// Number of tile vertically in the CGRA
47 static auto constexpr y_size = layout::y_max + 1;
48
49 /// Total number of tiles in the CGRA
50 static auto constexpr size = x_size*y_size;
51
52
53 /** A tuple with the coordinate tuples of all the tiles inside the
54 CGRA geometry */
55 static auto constexpr tile_coordinates = boost::hana::cartesian_product(
56 boost::hana::make_tuple(
57 boost::hana::range_c<int, y_min, y_size>
58 , boost::hana::range_c<int, x_min, x_size>
59 )
60 );
61
62
63 /** Generate a tuple by tiling the Tileable object onto the CGRA geometry
64
65 \param Tileable is the type to tile according to coordinate \p X
66 and \p Y
67 */
68 template <template <int X, int Y> typename Tileable>
69 static auto generate_tiles() {
70 return boost::hana::transform(
71 tile_coordinates
72 , [] (auto coord) {
73 return Tileable<boost::hana::at_c<1>(coord),
74 boost::hana::at_c<0>(coord)> {};
75 }
76 );
77 }

226 /** Test if a tile owns the start of the cascade_stream
227
228 \param[in] x is the horizontal tile coordinate

243 static bool constexpr is_cascade_end(int x, int y) {
244 // The x position depends on the parity of the last line
245 return x == ((y & 1) ? x_min : layout::x_max) && y == layout::y_max;

249 /** Compute the linearized coordinate of a tile along the cascade
250 stream
251
252 \param[in] x is the horizontal tile coordinate
253
254 \param[in] y is the vertical tile coordinate
255 */
256 static auto constexpr cascade_linear_id(int x, int y) {
257 return x_size*y + ((y & 1) ? layout::x_max - x : x);
258 }

261 /** Compute the horizontal coordinate of a tile from a cascade
262 linear id
263
264 \param[in] linear_id is a linear id along the cascade stream
265 */
266 static auto constexpr cascade_linear_x(int linear_id) {
267 auto x = linear_x(linear_id);
268 // According to the row parity, symmetrize the x coordinate
269 return (cascade_linear_y(linear_id) & 1) ? layout::x_max - x : x;
270 }

279 // Same as the linear id, actually

284 /** Test if a memory module coordinate offset is plausible
285
286 1 of the displacement is 0 and the other one is -1 or 1
287
288 \param[in] dx is the horizontal displacement (-1,0,+1)
289
290 \param[in] dy is the vertical displacement (-1,0,+1)
291 */
292 static bool constexpr is_valid_memory_module_offset(int dx, int dy) {
293 if (dx == 0)
294 // Cannot use std::abs() because it is not constexpr in C++17
295 return dy == -1 || dy == 1;
296 else if (dy == 0)
297 return dx == -1 || dx == 1;
298 else return false;
299 }
300
301
302 /** Test if a memory module exists and is connected to a tile
303
304 \param[in] x is the horizontal tile coordinate
305
306 \param[in] y is the vertical tile coordinate
307
308 \param[in] dx is the horizontal displacement (-1,0,+1)
309
310 \param[in] dy is the vertical displacement (-1,0,+1)
311 */
312 static bool constexpr is_memory_module(int x, int y, int dx, int dy) {
313 if (!is_valid_memory_module_offset(dx, dy))
314 return false;
315 if (y == y_min && dy == -1)
316 // No memory module on the South of the first (Southern) line
317 return false;
318 else if (y == layout::y_max && dy == 1)
319 // No memory module on the North of the last (Northern) line
320 return false;
321 else if (x == x_min && !(y & 1) && (dx == -1))
322 // No memory module on the West of the first (Western) column on
323 // even lines
324 return false;
325 else if (x == layout::x_max && (y & 1) && (dx == 1))
326 // No memory module on the East of the last (Eastern) column on
327 // odd lines
328 return false;
329 else return true;
330 }
331
332
333 /** Compute the memory module x coordinate delta from its tile
334
335 \param[in] y is the vertical tile coordinate
336
337 \param[in] dx is the horizontal displacement (-1,0,+1)
338 */
339 static auto constexpr memory_module_dx_coordinate(int y, int dx) {
340 // Mimic Figure 2-20 on p. 65
341 return (dx - 1)/2 + (y & 1);
342 }
343
344
345 /** Compute the linear id of a memory module attached to a tile
346
347 \param[in] x is the horizontal tile coordinate
348
349 \param[in] y is the vertical tile coordinate
350
351 \param[in] dx is the horizontal displacement (-1,0,+1)
352
353 \param[in] dy is the vertical displacement (-1,0,+1)
354 */
355 static auto constexpr memory_module_linear_id(int x, int y, int dx, int dy) {
356 if (dx == 0)
357 return linear_id(x, y + dy);
358 else
359 // Take into account the horizontal chessboard-like skewing
360 return linear_id(x + memory_module_dx_coordinate(y, dx), y);
361 }
