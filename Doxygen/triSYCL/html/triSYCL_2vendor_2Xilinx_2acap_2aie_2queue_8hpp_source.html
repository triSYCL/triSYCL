41 /** Provide a wait() function that actually can throw an exception
42 instead of ignoring it by the std::future::wait() */
43 struct queue_waiter : std::future<void> {
44 void wait() {
45 // Just call the function unwrapping the promise, it will
46 // rethrow if there is any exception pending
47 get();
48 }

51 /** Run synchronously a program execution on this queue
52
53 \param Tile is the description of the program tiles to
54 instantiate. By default each tile will run an empty program.
55
56 \param Memory is the description of the machine memory modules. By
57 default the machine has empty memory modules.
58 */
59 template <template <typename Device, int X, int Y>
60 typename Tile = acap::aie::tile,
61 template <typename Device, int X, int Y>
62 typename Memory = acap::aie::memory>
63 void run() const {

65 }

67 /** Run synchronously an heterogeneous invocable on this queue
68
69 \param f is an invocable taking an heterogeneous tile handler
70 */
71 template <typename Invocable> void run(Invocable&& f) const {
72 program<device> { aie_d }.run(std::forward<Invocable>(f));
73 }
74
75 /// Wait for all the device tiles on this queue to finish
76 void wait() const {
77 aie_d.for_each_tile([](auto& t) { t.wait(); });
78 }

80 /** Run synchronously a uniform invocable on this queue

84 template <typename Invocable> void uniform_run(Invocable&& f) const {
85 aie_d.for_each_tile(
86 [work = std::forward<Invocable>(f)](auto& t) {
