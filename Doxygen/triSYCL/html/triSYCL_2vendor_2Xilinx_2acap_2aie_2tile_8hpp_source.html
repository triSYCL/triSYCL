25/** The AI Engine tile infrastructure defining the program of a tile
26
27 This is the type you need to inherit from to define the program of
28 a CGRA tile.
29
30 \param AIE_Program is the type representing the full CGRA with the
31 programs and memory contents
32
33 \param X is the horizontal coordinate of the memory module
34
35 \param Y is the vertical coordinate of the memory module
36
37 \todo (Andrew Gozillon) FYI, until Intel fix the way they handle
38 base classes in the SYCL compiler frontend this tile_base can
39 never have non-default constructed data (on the device side
40 implementation, the host should be fine to have whatever). As the
41 hack I have in place just now only outlines a call to the base
42 classes default constructor, it might be feasible to work around
43 that though by calling a base class "fill" function in the tile
44 constructor or tweaking the way the run function of a tile is
45 outlined so that it outlines it as a copy of the full object
46 rather than components of the object and reconstructing it inside
47 the kernel (although you'll fall into the standard layout trap as
48 you'll be passing a full object to the kernel invocation and it'd
49 take some reworking of the Tile's main generation).
50
51 A proper fix for this is probably quite difficult as you'd need to
52 take into account the base classes dependencies among-st other
53 things like the constructor body not being available in the same
54 TU, some discussion here: https://github.com/intel/llvm/issues/488
55*/

268 /** Get the memory module relative to the tile

283 "Note that since a core tile has a 4-neighbor connectivity,"

285 " 1 or -1");
286 if constexpr (Dx == -1)
287 return mem_west();
288 else if constexpr (Dx == 1)
289 return mem_east();

352 /** An horizontal barrier using a lock
353
354 Implement a barrier across the tiles a line.
355
356 \param[in] lock is the ME lock to use. The lock 14 is used by
357 default
358 */
359 void horizontal_barrier(int lock = 14) {
360 if constexpr (y & 1) {
361 // Propagate a token from West to East and back
362 if constexpr (!is_west_column()) {
363 // Wait for the Western neighbour to be ready
364 mem().lock(lock).acquire_with_value(true);
365 }
366 if constexpr (is_memory_module_east()) {
367 mem_east().lock(lock).acquire_with_value(false);
368 // Unleash the Eastern neighbour
369 mem_east().lock(lock).release_with_value(true);
370 // Wait for the Eastern neighbour to acknowledge
371 mem_east().lock(lock).acquire_with_value(false);
372 }
373 if constexpr (!is_west_column()) {
374 // Acknowledge to the Western neighbour
375 mem().lock(lock).release_with_value(false);
376 }
377 } else {
378 // Propagate a token from East to West and back
379 if constexpr (!is_east_column()) {
380 // Wait for the Eastern neighbour to be ready
381 mem().lock(lock).acquire_with_value(true);
382 }
383 if constexpr (is_memory_module_west()) {
384 mem_west().lock(lock).acquire_with_value(false);
385 // Unleash the Western neighbour
386 mem_west().lock(lock).release_with_value(true);
387 // Wait for the Western neighbour to acknowledge
388 mem_west().lock(lock).acquire_with_value(false);
389 }
390 if constexpr (!is_east_column()) {
391 // Acknowledge to the Eastern neighbour
392 mem().lock(lock).release_with_value(false);
393 }
394 }
395 }

398 /** A vertical barrier using a lock
399
400 Implement a barrier across the tiles a line.
401
402 \param[in] lock is the ME lock to use. The lock 15 is used by
403 default
404 */
405 void vertical_barrier(int lock = 15) {
406 // Propagate a token from South to North and back
407 if constexpr (!is_south_row()) {
408 // Wait for the Southern neighbour to be ready
409 mem().lock(lock).acquire_with_value(true);
410 }
411 if constexpr (is_memory_module_north()) {
412 mem_north().lock(lock).acquire_with_value(false);
413 // Unleash the Northern neighbour
414 mem_north().lock(lock).release_with_value(true);
415 // Wait for the Northern neighbour to acknowledge
416 mem_north().lock(lock).acquire_with_value(false);
417 }
418 if constexpr (!is_south_row()) {
419 // Acknowledge to the Southern neighbour
420 mem().lock(lock).release_with_value(false);
421 }
422 }
