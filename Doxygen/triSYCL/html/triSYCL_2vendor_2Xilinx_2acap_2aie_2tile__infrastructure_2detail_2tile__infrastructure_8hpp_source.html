4/** \file
5
6 The basic AI Engine homogeneous tile, with the common
7 infrastructure to all the tiles, i.e. independent of x & y

10 This tile can be seen as the raw CGRA subdevice to run elemental
11 functions.
12
13 This is owned by the device, so for example the AXI stream switch

43/** The AI Engine tile infrastructure common to all the tiles

99 /** Map the user input port number to the AXI stream switch port
100
101 \param[in] port is the user port to use

103 static auto translate_input_port(int port) {
104 return axi_ss_t::translate_port(port, spl::me_0, spl::me_last,

118 /** Start the tile infrastructure associated to the AIE device tile
119
120 \param[in] x is the horizontal coordinate for this tile
121
122 \param[in] y is the vertical coordinate for this tile
123
124 \param[in] dev is the aie::detail::device used to control
125 hardware when using real hardware and provide some debug
126 information from inside the tile_infrastructure. Use auto
127 concept here to avoid explicit type causing circular dependency
128
129 \param[in] fiber_executor is the executor used to run
130 infrastructure details
131 */
132 tile_infrastructure(int x, int y, auto& dev,
133 ::trisycl::detail::fiber_pool& fiber_executor)
134 : x_coordinate { x }
135 , y_coordinate { y }
136 , dev { &dev }
137 , mi { dev }
138#if TRISYCL_XILINX_AIE_TILE_CODE_ON_FIBER
139 , fe { &fiber_executor }
140#endif
141 {
142 // Connect the core receivers to its AXI stream switch
143 for (auto p : views::enum_type(mpl::me_0, mpl::me_last))
144 output(p) =
145 std::make_shared<port_receiver<axi_ss_t>>(axi_ss, "core_receiver");
146 axi_ss.start(x, y, fiber_executor);
147 /* Create the core tile receiver DMAs and make them directly the
148 switch output ports */
149 for (auto p : axi_ss_geo::m_dma_range)
150 output(p) =
151 std::make_shared<receiving_dma<axi_ss_t>>(axi_ss, fiber_executor);
152 /* Create the core tile sender DMAs and connect them internally to
153 their switch input ports */
154 for (const auto& [d, p] :
155 ranges::views::zip(tx_dmas, axi_ss_geo::s_dma_range))
156 d.emplace(fiber_executor, input(p));
157 }

210 /** Get access to a receiver DMA

213 auto& rx_dma(int id) {
214 /** The output of the switch is actually a receiving DMA, so we
215 can view it as a DMA */
216 return static_cast<receiving_dma<axi_ss_t>&>(*output(
217 axi_ss_t::translate_port(id, mpl::dma_0, mpl::dma_last,
218 "The receiver DMA port is out of range")));

224 auto& tx_dma(int id) { return *tx_dmas.at(id); }

244 /// Launch an invocable on this tile
245 template <typename Work> void single_task(Work&& f) {
246 if (future_work.valid())
247 throw std::logic_error("Something is already running on this tile!");
248 // Launch the tile program immediately on a new executor engine
249#if TRISYCL_XILINX_AIE_TILE_CODE_ON_FIBER
250 future_work = fe->submit(std::forward<Work>(f));
251#else
252 future_work = std::async(std::launch::async, std::forward<Work>(f));
253#endif
254 }
255
256 /// Wait for the execution of the callable on this tile
257 void wait() { if (future_work.valid())
258 future_work.get();
259 }

267 /// Compute the size of the graphics representation of the processor
268 static vec<int, 2> display_core_size() {
269 // This is the minimum rectangle fitting all the processor outputs & inputs
270 return { 1 + ranges::distance(axi_ss_geo::m_me_range),
271 1 + ranges::distance(axi_ss_geo::s_me_range) };
272 }
273
274 /// Compute the size of the graphics representation of the tile
275 static vec<int, 2> display_size() {
276 // Just the sum of the size of its content
277 return display_core_size() + axi_ss_t::display_size();
278 }

280 /// Display the tile to a LaTeX context
281 void display(latex::context& c) const {
282 auto get_tikz_coordinate = [&](auto x, auto y) {
283 auto const [x_size, y_size] = display_size();
284 return (boost::format { "(%1%,%2%)" }
285 // Scale real LaTeX coordinate to fit
286 % c.scale(x_coordinate * x_size + x) %
287 c.scale(y_coordinate * y_size + y))
288 .str();
289 };
290 c.add((boost::format { " \\begin{scope}[name prefix = TileX%1%Y%2%]" } %
291 x_coordinate % y_coordinate)
292 .str());
293 axi_ss.display(c, display_core_size(), get_tikz_coordinate);
294
295 // Connect the core receivers to its AXI stream switch
296 for (auto [i, p] : axi_ss_geo::m_me_range | ranges::views::enumerate) {
297 c.add(
298 (boost::format { R"(
299 \coordinate(CoreIn%1%) at %2%;
300 \node[rotate=90,anchor=east](CoreIn%1%Label) at %2% {in(%1%)};
301 \draw (node cs:name=MMe%1%)
302 -| (node cs:name=CoreIn%1%);)" } %
303 i %
304 get_tikz_coordinate(i, ranges::distance(axi_ss_geo::m_me_range) + 1))
305 .str());
306 };
307 // Connect the core senders to its AXI stream switch
308 for (auto [i, p] : axi_ss_geo::s_me_range | ranges::views::enumerate) {
309 c.add(
310 (boost::format { R"(
311 \coordinate(CoreOut%1%) at %2%;
312 \node[anchor=east](CoreOut%1%Label) at %2% {out(%1%)};
313 \draw (node cs:name=CoreOut%1%)
314 -| (node cs:name=SMe%1%);)" } %
315 i %
316 get_tikz_coordinate(ranges::distance(axi_ss_geo::s_me_range), i + 1))
317 .str());
318 };
319 c.add((boost::format { R"(
320 \node[black] () at %1% {\texttt{tile<%2%,%3%>}};
321 \begin{scope}[on background layer]
322 \node [fill=orange!30, fit={(node cs:name=CoreIn0Label)
323 (node cs:name=CoreOut0Label)}]
324 (Core) {};
325 \end{scope}
326 \end{scope}
327
328)" } % get_tikz_coordinate(1, 0) %
329 x_coordinate % y_coordinate)
330 .str());
331 }
