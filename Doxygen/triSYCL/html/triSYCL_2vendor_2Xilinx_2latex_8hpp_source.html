54 /** Create a graphics context
55
56 \param[in] s is the 2D size of the drawing to be generated in mm
57
58 \param[in] f_name is the file name of the LaTeX to be generated
59 into or standard output by default
60 */
61 context(const vec<int, 2>& s, const std::string& f_name = {})
62 : size { s }, file_name { f_name } {
63 // Compute the scaling factor to remain under the TeX limit
64 auto m = std::max(size.x(), size.y());
65 scaling_factor = m > max_size ? max_size/m : 1;
66 // Generate the LaTeX header
67 out = (boost::format { R"(%% To be compiled with lualatex instead of pdflatex
68%% to avoid a bug on _ and to handle huge memory automatically.
69\documentclass{article}
70%% Use maximum of the page surface
71\usepackage[paperwidth=%1%mm,paperheight=%2%mm,top=0mm,bottom=0mm,
72 left=0mm,right=0mm]{geometry}
73%% Use a font allowing arbitrary size
74\usepackage{lmodern}
75%% The turbo-charged graphics package
76\usepackage{tikz}
77\usetikzlibrary{backgrounds,calc,decorations.pathmorphing,fit,patterns,mindmap}
78\usepackage{tikzlings}
79\definecolor{orangeSYCL}{RGB}{242,104,34}
80%% Some cool palettes
81\usepackage{xcolor-material}
82\usepackage{xcolor-solarized}
83%% Consider '_' as a (almost) normal character
84\usepackage[strings]{underscore}
85
86\begin{document}
87%% No page number, header or footer
88\thispagestyle{empty}
89%% Skip the usual space at the beginning of a paragraph
90\noindent
91%% Use a super small font. Use sans-serif for readability
92\fontsize{%3%}{%3%}\selectfont\sffamily
93%% Use remembering in every picture so we can use named coordinates
94%% across them
95\tikzstyle{every picture}+=[remember picture]
96\begin{tikzpicture}[%% Scale by 0.1, so the unit is 1mm instead of default 1cm
97 scale = 0.1,
98 %% Default style
99 gray,
100 style = {line width = %4%mm, ->}]
101
102)" } % scale(size.x()) % scale(size.y()) % scale(1) % scale(0.01)).str();
103 }

146 /** Build a TikZ/PGF valid name string by concatenating a prefix and a
147 capitalized node name after having removed any '_'
148
149 \input[in] node_name is a string-like TikZ/PGF that may contain some '_'
150
151 \input[in] prefix is an optional string-like prefix
152 */
153 static std::string clean_node(const std::string_view& node_name,
154 const std::string_view& prefix = "") {
155 // All the characters but the '_'
156 auto without_ = ranges::views::remove(node_name, '_');
157 return ranges::views::concat
158 (ranges::views::all(prefix),
159 // Take the first character if any and upper-case it
160 without_ | ranges::views::take(1)
161 | ranges::views::transform([] (auto c) { return std::toupper(c); }),
162 // And take all the character after the first one if any
163 ranges::views::drop(without_, 1))
164 | ranges::to<std::string>;
165 }
