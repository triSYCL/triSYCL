73 /** Flattening helper that does not change scalar values but flatten a
74 vec<T, n> v into a tuple<T, T,..., T>{ v[0], v[1],..., v[n-1] }
75
76 If we have a vector, just forward its array content since an array
77 has also a tuple interface :-) (23.3.2.9 Tuple interface to class

117 /* apply_unary_functor_impl & apply_unary_functor generate a sequence from
118 a passed in vector and then execute a function across the range of the
119 vector returning the result as a new vector.
120 */
121 template <typename T, typename F, std::size_t... Is>
122 static auto apply_unary_functor_impl(T tuple, F f,

124 return ::trisycl::vec<DataType, NumElements>{f(std::get<Is>(tuple))...};

226 // Applies a binary function across each element of the calling vector and the
227 // passed in vector x to generate a new vector. It leaves both the original
228 // vectors unaltered.
229 //
230 // Note: could in theory be made less restrictive and allow zipping with a
231 // scalar type for x and allow zipping across two differing sized vectors.
232 // It would require an extra apply_binary_functor overload and a test to find
233 // the smallest index_sequence between the calling vector and x.
234 template <typename T, int size, typename F>
235 auto zip(::trisycl::vec<T, size> x, F f) const {
236 static_assert(size == NumElements,
237 "zip currently does not support vec's of differing element size");
238 return apply_binary_functor(*this, x, f);
239 }
