566/* Provide a tuple-like interface to the SYCL vec with some type
567 traits and function template specialization in std::, besides the
568 get<> member function.
569
570 This is useful to have a data type compatible with structure binding
571*/
572
573/// The std::tuple_element specialization for vec
574template<std::size_t I, typename T, int Size>
575struct tuple_element<I, trisycl::vec<T, Size>> {
576 static_assert(I < Size, "The vec<> has not enough elements!");
577 // All the elements have the same type
578 using type = T;
579};
580
581
582/// The std::tuple_size specialization for vec
583template <typename T, int Size>
584struct tuple_size<trisycl::vec<T, Size>>
585 : integral_constant<size_t, Size>
586{ };
587
588
589/** Provide a get<> accessor to have C++17 structured binding working
590
591 Note that it works without this with current std::array based
592 implementation on the host device but we prefer to have a more
593 realistic const binding
594*/
595template <int I, typename T, int Size>
596auto get(const trisycl::vec<T, Size>& v) {
597 static_assert(I < Size, "The vec<> has not enough elements!");
598 return v[I];
599}
